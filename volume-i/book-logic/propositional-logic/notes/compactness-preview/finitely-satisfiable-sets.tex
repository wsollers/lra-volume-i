\begin{definitionbox}{Definition (Finite Subset of a Set of Propositional Formulas)}
\begin{definition}[Finite Subset of a Set of Propositional Formulas]
\label{def:finite-subset-of-set-of-propositional-formulas}
Let \(\Gamma\subseteq\WFF\). A finite subset of \(\Gamma\) is a set
\(\Delta\subseteq\Gamma\) containing only finitely many formulas.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\Delta\text{ is a finite subset of }\Gamma
\Longleftrightarrow
\Delta\subseteq\Gamma\land \Delta\text{ is finite}.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\Delta\subseteq\Gamma
\quad\text{and}\quad
\Delta\text{ is finite}
\]
means that \(\Delta\) is a finite selection of formulas from \(\Gamma\).
\end{remark*}

\begin{remark*}[Interpretation]
A finite subset is a finite test case cut out of a larger set of formulas.
\end{remark*}

\begin{definitionbox}{Definition (Finitely Satisfiable Set of Propositional Formulas)}
\begin{definition}[Finitely Satisfiable Set of Propositional Formulas]
\label{def:finitely-satisfiable-set-of-propositional-formulas}
Let \(\Gamma\subseteq\WFF\). The set \(\Gamma\) is finitely satisfiable if every
finite subset of \(\Gamma\) is satisfiable.
\[
\operatorname{FinitelySatisfiableSet}(\Gamma)
\Longleftrightarrow
\forall\Delta\,\Bigl((\Delta\subseteq\Gamma\land \Delta\text{ is finite})
\Rightarrow \operatorname{SatisfiableSet}(\Delta)\Bigr).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\Gamma\text{ is finitely satisfiable}
\Longleftrightarrow
\forall\Delta\,\Bigl((\Delta\subseteq\Gamma\land \Delta\text{ is finite})
\Rightarrow \exists v\,(\operatorname{TruthAssignment}(v)\land v\models\Delta)\Bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{FinitelySatisfiableSet}(\Gamma)
\]
means that every finite portion of \(\Gamma\) has at least one satisfying truth
assignment.
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\neg\operatorname{FinitelySatisfiableSet}(\Gamma)
\Longleftrightarrow
\exists\Delta\,\Bigl(\Delta\subseteq\Gamma\land \Delta\text{ is finite}\land
\operatorname{UnsatisfiableSet}(\Delta)\Bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{FinitelySatisfiableSet}(\Gamma)
\]
means that some finite subset of \(\Gamma\) already has no satisfying truth
assignment.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.] Finite satisfiability fails when at least one finite subset of \(\Gamma\) is
unsatisfiable.
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.] \[
\neg\operatorname{FinitelySatisfiableSet}(\Gamma)
\Longleftrightarrow
\exists\Delta\,\Bigl(\Delta\subseteq\Gamma\land \Delta\text{ is finite}\land
\operatorname{UnsatisfiableSet}(\Delta)\Bigr).
\]
So failure of finite satisfiability is witnessed by one finite unsatisfiable
subcollection.
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Finite satisfiability says that no finite test has found a contradiction. It
does not name a single truth assignment for all of \(\Gamma\); it only says
that each finite test can be passed.
\end{remark*}
