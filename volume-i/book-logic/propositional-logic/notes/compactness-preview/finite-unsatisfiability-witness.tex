\begin{definitionbox}{Definition (Unsatisfiability Witness)}
\begin{definition}[Unsatisfiability Witness]
\label{def:unsatisfiability-witness-propositional-logic}
Let \(\Gamma\subseteq\WFF\). A subset \(\Delta\subseteq\Gamma\) is an
unsatisfiability witness for \(\Gamma\) if \(\Delta\) is unsatisfiable.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\Delta\text{ is an unsatisfiability witness for }\Gamma
\Longleftrightarrow
\Delta\subseteq\Gamma\land \operatorname{UnsatisfiableSet}(\Delta).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\Delta\subseteq\Gamma
\quad\text{and}\quad
\operatorname{UnsatisfiableSet}(\Delta)
\]
means that the formulas in \(\Delta\) already explain why the larger set
\(\Gamma\) cannot be satisfied, provided \(\Delta\) is part of \(\Gamma\).
\end{remark*}

\begin{remark*}[Interpretation]
An unsatisfiability witness is a subcollection that already fails to be
jointly satisfiable.
\end{remark*}

\begin{definitionbox}{Definition (Finite Unsatisfiability Witness)}
\begin{definition}[Finite Unsatisfiability Witness]
\label{def:finite-unsatisfiability-witness-propositional-logic}
Let \(\Gamma\subseteq\WFF\). A subset \(\Delta\subseteq\Gamma\) is a finite
unsatisfiability witness for \(\Gamma\) if \(\Delta\) is finite and
unsatisfiable.
\[
\operatorname{FiniteUnsatisfiabilityWitness}(\Delta,\Gamma)
\Longleftrightarrow
\Delta\subseteq\Gamma\land \Delta\text{ is finite}\land
\operatorname{UnsatisfiableSet}(\Delta).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\Delta\text{ is a finite unsatisfiability witness for }\Gamma
\Longleftrightarrow
\Delta\subseteq\Gamma\land \Delta\text{ is finite}\land
\operatorname{UnsatisfiableSet}(\Delta).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{FiniteUnsatisfiabilityWitness}(\Delta,\Gamma)
\]
means that \(\Delta\) is a finite subset of \(\Gamma\) and that \(\Delta\)
already has no satisfying truth assignment.
\end{remark*}

\begin{remark*}[Interpretation]
A finite unsatisfiability witness is a finite obstruction. It says that the
failure of \(\Gamma\) is already visible before one inspects all formulas in
\(\Gamma\).
\end{remark*}

\begin{theorembox}{Theorem (Finite Unsatisfiability Witness)}
\begin{theorem}[Finite Unsatisfiability Witness]
\label{thm:finite-unsatisfiability-witness-propositional-logic}
Let \(\Gamma\subseteq\WFF\). If there exists a finite
\(\Delta\subseteq\Gamma\) such that \(\Delta\) is unsatisfiable, then
\(\Gamma\) is unsatisfiable.

\medskip
\noindent\hyperref[prf:finite-unsatisfiability-witness-propositional-logic]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\exists\Delta\,\operatorname{FiniteUnsatisfiabilityWitness}(\Delta,\Gamma)
\Rightarrow
\operatorname{UnsatisfiableSet}(\Gamma).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\exists\Delta\,\operatorname{FiniteUnsatisfiabilityWitness}(\Delta,\Gamma)
\Rightarrow
\operatorname{UnsatisfiableSet}(\Gamma)
\]
means that a single finite unsatisfiable subcollection forces the whole set
\(\Gamma\) to be unsatisfiable.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.] The theorem can fail only if a truth assignment satisfied \(\Gamma\) while a
finite subset \(\Delta\subseteq\Gamma\) remained unsatisfiable.
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.] \[
\operatorname{SatisfiableSet}(\Gamma)
\Rightarrow
\forall\Delta\,\bigl(\Delta\subseteq\Gamma\Rightarrow
\operatorname{SatisfiableSet}(\Delta)\bigr).
\]
If the whole set has a satisfying truth assignment, each subset inherits one.
\end{description}
\end{remark*}

\begin{remark*}[Contrapositive quantified statement]
\[
\operatorname{SatisfiableSet}(\Gamma)
\Rightarrow
\forall\Delta\,\bigl(\Delta\subseteq\Gamma\Rightarrow
\operatorname{SatisfiableSet}(\Delta)\bigr).
\]
\end{remark*}

\begin{remark*}[Contrapositive predicate reading]
\[
\operatorname{SatisfiableSet}(\Gamma)
\Rightarrow
\operatorname{FinitelySatisfiableSet}(\Gamma)
\]
means that one truth assignment for all of \(\Gamma\) also satisfies every
finite subset of \(\Gamma\).
\end{remark*}

\begin{remark*}[Interpretation]
An unsatisfiable finite subset is decisive. Adding more formulas cannot restore
a truth assignment that already fails on the finite subset.
\end{remark*}

\begin{corollarybox}{Corollary (Finite Witness Contrapositive)}
\begin{corollary}[Finite Witness Contrapositive]
\label{cor:finite-unsatisfiability-witness-contrapositive-propositional-logic}
Let \(\Gamma\subseteq\WFF\). If \(\Gamma\) is satisfiable, then every finite
subset of \(\Gamma\) is satisfiable.

\medskip
\noindent\hyperref[prf:finite-unsatisfiability-witness-contrapositive-propositional-logic]{Go to proof.}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{SatisfiableSet}(\Gamma)
\Rightarrow
\operatorname{FinitelySatisfiableSet}(\Gamma).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{SatisfiableSet}(\Gamma)
\Rightarrow
\operatorname{FinitelySatisfiableSet}(\Gamma)
\]
means that full satisfiability automatically gives finite satisfiability.
\end{remark*}

\begin{remark*}[Interpretation]
This is the easy direction connecting full satisfiability to finite
satisfiability: one truth assignment that works for all formulas also works
for any finite selection.
\end{remark*}
