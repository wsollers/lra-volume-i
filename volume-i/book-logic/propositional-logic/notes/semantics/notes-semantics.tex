\begin{definitionbox}{Definition (Truth Assignment)}
\begin{definition}[Truth Assignment]
\label{def:truth-assignment-propositional-logic}
A \emph{truth assignment} is a function
\[
v:\Prop\to\mathbb{B}.
\]
For each propositional variable \(P\in\Prop\), the value \(v(P)\) is the truth
value assigned to \(P\).
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{TruthAssignment}(v)
\Longleftrightarrow
v:\Prop\to\mathbb{B}.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{TruthAssignment}(v)
\]
means that \(v\) assigns exactly one truth value to each propositional variable.
\end{remark*}

\begin{remark*}[Interpretation]
A truth assignment gives semantic values to the atomic formulas. The recursive
truth conditions for the connectives then determine the truth value of every
well-formed formula.
\end{remark*}

\begin{remark*}[Examples]
If \(\Prop=\{P_1,P_2,P_3,\ldots\}\), then a truth assignment may satisfy
\[
v(P_1)=\True,\qquad v(P_2)=\False,\qquad v(P_3)=\True,
\]
with values assigned to every remaining propositional variable.
\end{remark*}

\begin{remark*}[Non-Examples]
A partial assignment that gives values only to \(P_1\) and \(P_2\) is not a truth
assignment on \(\Prop\). It may be useful for local computations, but it is not a
function \(\Prop\to\mathbb{B}\).
\end{remark*}

\begin{definitionbox}{Definition (Extension of a Truth Assignment to Formulas)}
\begin{definition}[Extension of a Truth Assignment to Formulas]
\label{def:extended-truth-assignment-propositional-logic}
Let
\[
v:\Prop\to\mathbb{B}
\]
be a truth assignment. The \emph{extension} of \(v\) to well-formed formulas is
the function
\[
\widehat v:\WFF\to\mathbb{B}
\]
defined recursively as follows:
\[
\widehat v(P)=v(P)
\quad\text{for every }P\in\Prop,
\]
\[
\widehat v(\neg\varphi)=\True
\Longleftrightarrow
\widehat v(\varphi)=\False,
\]
\[
\widehat v(\varphi\land\psi)=\True
\Longleftrightarrow
\widehat v(\varphi)=\True
\text{ and }
\widehat v(\psi)=\True,
\]
\[
\widehat v(\varphi\lor\psi)=\True
\Longleftrightarrow
\widehat v(\varphi)=\True
\text{ or }
\widehat v(\psi)=\True,
\]
\[
\widehat v(\varphi\to\psi)=\False
\Longleftrightarrow
\widehat v(\varphi)=\True
\text{ and }
\widehat v(\psi)=\False,
\]
and
\[
\widehat v(\varphi\leftrightarrow\psi)=\True
\Longleftrightarrow
\widehat v(\varphi)=\widehat v(\psi).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
For every truth assignment \(v:\Prop\to\mathbb{B}\), its extension is a function
\[
\widehat v:\WFF\to\mathbb{B}
\]
satisfying the recursive truth clauses for propositional variables, negation,
conjunction, disjunction, material implication, and biconditional.
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{ExtendedTruthAssignment}(\widehat v,v)
\]
means that \(\widehat v\) is the recursively defined truth-value function on
\(\WFF\) induced by the atomic truth assignment \(v\).
\end{remark*}

\begin{remark*}[Interpretation]
The original assignment \(v\) gives truth values only to propositional variables.
The extension \(\widehat v\) evaluates every formula by following the formula's
syntactic construction.
\end{remark*}

\begin{remark*}[Examples]
\[
\begin{array}{c|c}
\varphi & \neg\varphi\\
\hline
\True & \False\\
\False & \True
\end{array}
\qquad
\begin{array}{c|c|c}
\varphi & \psi & \varphi\land\psi\\
\hline
\True & \True & \True\\
\True & \False & \False\\
\False & \True & \False\\
\False & \False & \False
\end{array}
\]
\[
\begin{array}{c|c|c}
\varphi & \psi & \varphi\lor\psi\\
\hline
\True & \True & \True\\
\True & \False & \True\\
\False & \True & \True\\
\False & \False & \False
\end{array}
\qquad
\begin{array}{c|c|c}
\varphi & \psi & \varphi\to\psi\\
\hline
\True & \True & \True\\
\True & \False & \False\\
\False & \True & \True\\
\False & \False & \True
\end{array}
\]
\[
\begin{array}{c|c|c}
\varphi & \psi & \varphi\leftrightarrow\psi\\
\hline
\True & \True & \True\\
\True & \False & \False\\
\False & \True & \False\\
\False & \False & \True
\end{array}
\]
\end{remark*}

\begin{remark*}[Examples]
The truth tables are not additional formation rules. They specify how each
primitive connective acts on truth values once the formula has already been
formed syntactically.
\end{remark*}

\begin{theorembox}{Theorem (Unique Extension of a Truth Assignment)}
\begin{theorem}[Unique Extension of a Truth Assignment]
\label{thm:unique-extension-truth-assignment-propositional-logic}
Every truth assignment
\[
v:\Prop\to\mathbb{B}
\]
has a unique extension
\[
\widehat v:\WFF\to\mathbb{B}
\]
satisfying the recursive truth clauses for propositional variables and the
primitive connectives.

\noindent\hyperref[prf:unique-extension-truth-assignment-propositional-logic]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
(\forall v:\Prop\to\mathbb{B})(\exists!\widehat v:\WFF\to\mathbb{B})
\]
such that \(\widehat v\) agrees with \(v\) on \(\Prop\) and satisfies the
recursive truth clauses for
\[
\neg,\land,\lor,\to,\leftrightarrow.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{TruthAssignment}(v)
\Longrightarrow
\exists!\widehat v\,
\operatorname{ExtendedTruthAssignment}(\widehat v,v).
\]
\end{remark*}

\begin{remark*}[Interpretation]
This theorem guarantees that semantic evaluation is well-defined. Once the
truth values of the propositional variables are fixed, every well-formed formula
has exactly one truth value.
\end{remark*}

if
\[
\widehat v(\varphi)=\mathrm{True}.

\end{definition}

\Longleftrightarrow
\widehat v(\varphi)=\mathrm{False}.

\begin{definitionbox}{Definition (Truth Table)}
\begin{definition}[Truth Table]
\label{def:truth-table-propositional-logic}
Let \(\varphi\in\WFF\) and suppose the variables occurring in \(\varphi\) are among \(P_1,\ldots,P_n\). A truth table for \(\varphi\) is a finite table with one row for each assignment
\[
a:\{P_1,\ldots,P_n\}\to\mathbb{B}
\]
and a final column recording the value of \(\widehat v(\varphi)\) for any truth assignment \(v\) extending \(a\).
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{TruthTable}(\varphi)
\Longleftrightarrow
\varphi\in\WFF
\land
\operatorname{Var}(\varphi)=\{P_1,\ldots,P_n\}
\land
\text{the table has }2^n\text{ assignment rows}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A truth table is finite because every formula contains only finitely many variables. It records the complete truth behavior of a formula.
\end{remark*}

\begin{definition}[Logical Consequence]

\[

\]

\end{definition}

\[
\Gamma\models\varphi
\Longleftrightarrow
\Gamma\subseteq\WFF
\land
\varphi\in\WFF
\land
(\forall v:\Prop\to\mathbb{B})

v\models\Gamma\to v\models\varphi
\right].
\]

\[
\operatorname{LogicalConsequence}(\Gamma,\varphi)

v\models\Gamma
\land
v\not\models\varphi

v|_{\operatorname{Var}(\varphi)}

w|_{\operatorname{Var}(\varphi)}

Truth tables are finite because every formula contains only finitely many propositional variables.

\begin{definitionbox}{Definition (Boolean Function)}
\begin{definition}[Boolean Function]
\label{def:boolean-function-propositional-logic}
An \(n\)-ary Boolean function is a function
\[
f:\mathbb{B}^n\to\mathbb{B}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
BooleanFunction_n(f)
\Longleftrightarrow
f:\mathbb{B}^n\to\mathbb{B}.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
BooleanFunction_n(f)
\]
means that \(f\) takes \(n\) truth values as input and returns one truth value.
\end{remark*}

\begin{remark*}[Interpretation]
Boolean functions are the abstract truth-functions that formulas may represent.
\end{remark*}

\begin{remark*}[Exposition]
An input to an \(n\)-ary Boolean function is a tuple
\((b_1,\ldots,b_n)\in\mathbb{B}^n\). These tuples are the rows of the truth table, and there are \(2^n\) of them. The Boolean function is not a single row; it is the full rule assigning exactly one truth value in \(\mathbb{B}\) to every row.
\end{remark*}
