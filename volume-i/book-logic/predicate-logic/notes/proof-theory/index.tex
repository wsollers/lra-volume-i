\begin{toolkitbox}{Quantifier Inference Rules --- Quick Reference}
\small
\begin{tabular}{l p{0.68\linewidth}}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[def:ui]{UI}   & $\forall x\,\varphi \Rightarrow \varphi[t/x]$; $t$ free for $x$ \\
\hyperref[def:ug]{UG}   & $\varphi \Rightarrow \forall x\,\varphi$; $x$ not free in assumptions \\
\hyperref[def:ei]{EI}   & $\exists x\,\varphi \Rightarrow \varphi[c/x]$; $c$ fresh witness \\
\hyperref[def:eg]{EG}   & $\varphi[t/x] \Rightarrow \exists x\,\varphi$ \\
\bottomrule
\end{tabular}
\end{toolkitbox}

\begin{toolkitbox}{Quantifier Proof Strategies --- Quick Reference}
\small
\begin{tabular}{l p{0.68\linewidth}}
\toprule
\textbf{Name} & \textbf{Meaning} \\
\midrule
Prove $\forall x\,\varphi(x)$  & Fix an arbitrary $x$; prove $\varphi(x)$ for that $x$; use UG \\
Prove $\exists x\,\varphi(x)$  & Produce a specific witness $t$; verify $\varphi(t)$; use EG \\
Use $\forall x\,\varphi(x)$    & Instantiate at any term $t$ to get $\varphi(t)$; use UI \\
Use $\exists x\,\varphi(x)$    & Introduce a fresh name $c$ for the witness; use EI \\
Negate $\forall x\,\varphi$    & Convert to $\exists x\,\neg\varphi$ by a negation law \\
Negate $\exists x\,\varphi$    & Convert to $\forall x\,\neg\varphi$ by a negation law \\
\midrule
Prove $\forall x\,\exists y\,\varphi(x,y)$ & Fix $x$; then construct $y$ depending on $x$; use UG then EG \\
Prove $\exists y\,\forall x\,\varphi(x,y)$ & Find $y$ first; then verify for all $x$; use EG then UG \\
\bottomrule
\end{tabular}
\end{toolkitbox}

\begin{toolkitbox}{Equality Rules --- Quick Reference}
\small
\begin{tabular}{l p{0.68\linewidth}}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[def:eq-refl]{Reflexivity (EqI)}    & Primitive rule with schema $t = t$ \\
\hyperref[def:eq-elim]{Equality Elim (EqE)}  & Primitive rule: $t_1=t_2,\;\varphi[t_1/x] \Rightarrow \varphi[t_2/x]$ \\
\hyperref[thm:eq-sym]{Symmetry}             & Derived rule: $t_1=t_2 \Rightarrow t_2=t_1$ \\
\hyperref[thm:eq-trans]{Transitivity}         & Derived rule: $t_1=t_2,\;t_2=t_3 \Rightarrow t_1=t_3$ \\
\hyperref[thm:eq-term]{Term substitution}    & Derived rule for substituting equals inside terms \\
\hyperref[thm:eq-pred]{Predicate substitution}  & Derived rule for substituting equals inside predicates \\
\bottomrule
\end{tabular}
\end{toolkitbox}
