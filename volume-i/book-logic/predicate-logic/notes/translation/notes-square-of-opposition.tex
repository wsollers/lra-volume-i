\begin{definitionbox}{Definition (Universal Affirmative Form)}
\begin{definition}[Universal Affirmative Form]
\label{def:categorical-a-form-predicate-logic}
The \emph{universal affirmative} categorical form, called \(A\), translates
``All \(A\) are \(B\)'' as
\[
\forall x\,(A(x)\to B(x)).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\forall x\,(A(x)\to B(x)).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\(\varphi\in\mathsf{CategoricalAForm}(A,B)\) means that \(\varphi\) is the
universal affirmative translation from \(A\) to \(B\).
\end{remark*}

\begin{remark*}[Interpretation]
The \(A\)-form says that every object satisfying the subject predicate also
satisfies the predicate predicate. It does not by itself say that anything
satisfies the subject predicate.
\end{remark*}

\label{def:categorical-e-form-predicate-logic}
The \emph{universal negative} categorical form, called \(E\), translates
``No \(A\) are \(B\)'' as
\[
\forall x\,(A(x)\to \neg B(x)).
\]
\end{definition}
\end{definitionbox}

universal negative translation from \(A\) to \(B\).

\begin{propositionbox}{Proposition (A-O Contradiction)}
\begin{proposition}[A-O Contradiction]
\label{prop:a-o-contradictory-predicate-logic}
The \(A\)-form \(\forall x\,(A(x)\to B(x))\) and the \(O\)-form
\(\exists x\,(A(x)\land \neg B(x))\) are contradictory: exactly one of them is
true.

\noindent\hyperref[prf:a-o-contradictory-predicate-logic]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\neg\forall x\,(A(x)\to B(x))
\Longleftrightarrow
\exists x\,(A(x)\land \neg B(x)).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\(\operatorname{ContradictoryPair}(\varphi_A,\varphi_O)\) records that the
\(A\)-form and \(O\)-form cannot share a truth value.
\end{remark*}

\begin{remark*}[Interpretation]
The \(O\)-form is exactly the counterexample to the \(A\)-form: an object that
is \(A\) but not \(B\).
\end{remark*}

\begin{propositionbox}{Proposition (Contrariety Requires Existential Import)}
\begin{proposition}[Contrariety Requires Existential Import]
\label{prop:existential-import-contrariety-predicate-logic}
If \(\exists x\,A(x)\), then the \(A\)-form and \(E\)-form cannot both be true.

\noindent\hyperref[prf:existential-import-contrariety-predicate-logic]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\exists x\,A(x)
\to
\neg\bigl(
  \forall x\,(A(x)\to B(x))
  \land
  \forall x\,(A(x)\to \neg B(x))
\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\(\operatorname{ContraryWithExistentialImport}(\varphi_A,\varphi_E)\) records
that contrariety holds only after the subject predicate is known to have a
witness.
\end{remark*}

\begin{remark*}[Interpretation]
With an \(A\)-object available, the two universal claims force both \(B\) and
\(\neg B\) of that object. Without such an object, both universals may be
vacuously true. The traditional square of opposition assumes existential
import; modern first-order logic does not automatically assume that a universal
statement implies existence.
\end{remark*}
