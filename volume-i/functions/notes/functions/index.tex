\begin{toolkitbox}{Functions --- Quick Reference}
\small
\begin{tabular}{l l l}
\toprule
\textbf{Concept} & \textbf{Meaning} & \textbf{Detail} \\
\midrule
Functional relation & Each input has at most one output                         & \hyperref[def:functional-relation]{$\downarrow$ Def} \\
Total relation on $A$ & Each input has at least one output                       & \hyperref[def:total-relation-on-domain]{$\downarrow$ Def} \\
Function         & Left-total, right-unique relation                            & \hyperref[def:function]{$\downarrow$ Def} \\
Partial function & Functional, not necessarily total                            & \hyperref[def:partial-function]{$\downarrow$ Def} \\
Function equality & Same domain, codomain, and pointwise values                 & \hyperref[def:function-equality]{$\downarrow$ Def} \\
Many-place function & Functional relation in the last coordinate                 & \hyperref[def:many-place-functional-relation]{$\downarrow$ Def} \\
Domain           & $\dom(f)=A$ for $f:A\to B$                                 & \hyperref[def:domain]{$\downarrow$ Def} \\
Codomain         & $\cod(f)=B$ for $f:A\to B$                                  & \hyperref[def:codomain]{$\downarrow$ Def} \\
Image of function & $\im(f) = \{f(a) \mid a \in A\}$                            & \hyperref[def:image-function]{$\downarrow$ Def} \\
Image of set      & $f(S) = \{f(a) \mid a \in S\}$ for $S \subseteq A$          & \hyperref[def:image-set]{$\downarrow$ Def} \\
Preimage          & $f^{-1}(T) = \{a \in A \mid f(a) \in T\}$ for $T \subseteq B$ & \hyperref[def:preimage]{$\downarrow$ Def} \\
Fiber             & $f^{-1}(\{b\})$: preimage of a singleton                    & \hyperref[def:fiber]{$\downarrow$ Def} \\
Graph             & $\{(a,b) \in A\times B \mid b = f(a)\}$                     & \hyperref[def:graph]{$\downarrow$ Def} \\
Injective         & Distinct inputs $\Rightarrow$ distinct outputs               & \hyperref[def:injective]{$\downarrow$ Def} \\
Surjective        & Every codomain element is achieved                          & \hyperref[def:surjective]{$\downarrow$ Def} \\
Bijective         & Injective and surjective                                    & \hyperref[def:bijective]{$\downarrow$ Def} \\
\midrule
Function test     & Function iff total and functional                           & \hyperref[thm:function-total-functional-characterization]{$\downarrow$ Thm} \\
One-one test      & Injective iff converse graph is functional                  & \hyperref[prop:one-to-one-converse-functional]{$\downarrow$ Prop} \\
Bijective test    & Unique correspondence in both directions                    & \hyperref[prop:bijection-unique-both-directions]{$\downarrow$ Prop} \\
\midrule
Identity          & $\id_A(a) = a$                                              & \hyperref[def:identity]{$\downarrow$ Def} \\
Inclusion map     & $\iota : A \hookrightarrow B$, $\iota(a) = a$ for $A \subseteq B$ & \hyperref[def:inclusion]{$\downarrow$ Def} \\
Composition       & $(g \circ f)(a) = g(f(a))$                                  & \hyperref[def:composition]{$\downarrow$ Def} \\
Inverse           & Defined for bijections; $f^{-1} \circ f = \id_A$            & \hyperref[def:inverse]{$\downarrow$ Def} \\
Left inverse      & $g\circ f=\id_A$                                           & \hyperref[def:left-inverse-function]{$\downarrow$ Def} \\
Right inverse     & $f\circ h=\id_B$                                           & \hyperref[def:right-inverse-function]{$\downarrow$ Def} \\
Restriction       & $f|_C : C \to B$                                            & \hyperref[def:restriction]{$\downarrow$ Def} \\
Extension         & $g : A' \to B$ with $g|_A = f$                              & \hyperref[def:fn-extension]{$\downarrow$ Def} \\
Empty function    & Unique function $\varnothing\to B$                          & \hyperref[def:empty-function]{$\downarrow$ Def} \\
\bottomrule
\end{tabular}
\end{toolkitbox}
