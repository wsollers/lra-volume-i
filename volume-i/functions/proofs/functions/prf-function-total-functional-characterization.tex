\begin{theorem*}[Functional Relation Characterization of Functions]
Let $A$ and $B$ be sets, and let $R\subseteq A\times B$. Then $R$ is a
function from $A$ to $B$ if and only if $R$ is total on $A$ and functional
from $A$ to $B$.
\end{theorem*}
