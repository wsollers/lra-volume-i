\begin{signaturebox}{Signature Presentation: A Simple Arithmetic Language}
Let
\[
  \Sigma_{\mathrm{arith}}
  =
  \bigl(
    \mathcal{C}_{\mathrm{arith}},
    \mathcal{F}_{\mathrm{arith}},
    \mathcal{R}_{\mathrm{arith}},
    \operatorname{ar}
  \bigr)
\]
where
\[
  \mathcal{C}_{\mathrm{arith}}=\{0,1\},\qquad
  \mathcal{F}_{\mathrm{arith}}=\{+,\cdot,S\},\qquad
  \mathcal{R}_{\mathrm{arith}}=\{<\}.
\]
The arity function is specified by
\[
  \operatorname{ar}(0)=\operatorname{ar}(1)=0,\qquad
  \operatorname{ar}(S)=1,\qquad
  \operatorname{ar}(+)=\operatorname{ar}(\cdot)=2,\qquad
  \operatorname{ar}(<)=2.
\]

\begin{signaturetable}
$0,1$ & constants & $0$ & distinguished constant names \\
$S$ & function symbol & $1$ & successor operation symbol \\
$+,\cdot$ & function symbols & $2$ & binary operation symbols \\
$<$ & relation symbol & $2$ & binary order relation symbol \\
\end{signaturetable}

The signature records only the available non-logical symbols and their arities.
It does not assert any axioms about those symbols.
\end{signaturebox}

\begin{definition}[First-Order Signature]
\label{def:signature}
A \emph{first-order signature} is a tuple
\[
  \Sigma=(\mathcal{C},\mathcal{F},\mathcal{R},\operatorname{ar}),
\]
where $\mathcal{C}$ is a set of constant symbols, $\mathcal{F}$ is a set of
function symbols, $\mathcal{R}$ is a set of relation symbols, and
\[
  \operatorname{ar}:\mathcal{F}\cup\mathcal{R}\to \mathbb{Z}_{>0}
\]
assigns to each function or relation symbol its number of arguments.
\end{definition}

\begin{remark*}[Standard quantified statement]
For every $s\in\mathcal{F}\cup\mathcal{R}$, the value
$\operatorname{ar}(s)$ is a positive integer. If $s\in\mathcal{F}$, then
$s$ forms terms with $\operatorname{ar}(s)$ inputs; if $s\in\mathcal{R}$,
then $s$ forms atomic formulas with $\operatorname{ar}(s)$ inputs.
\end{remark*}

\begin{remark*}[Interpretation]
A signature is the formal vocabulary available before any axioms are chosen.
It says which symbols may occur and how many input places each symbol has.
\end{remark*}

\begin{definition}[Arity]
\label{def:arity}
Let $\Sigma=(\mathcal{C},\mathcal{F},\mathcal{R},\operatorname{ar})$ be a
first-order signature. The \emph{arity function} is the map
\[
  \operatorname{ar}:\mathcal{F}\cup\mathcal{R}\to\mathbb{Z}_{>0}
\]
that assigns to each function or relation symbol the number of arguments it
requires. When constants are treated as $0$-ary function symbols, the codomain
is enlarged to include $0$.
\end{definition}

\begin{remark*}[Standard quantified statement]
For every $s\in\mathcal{F}\cup\mathcal{R}$, there is a unique positive
integer $n$ such that $\operatorname{ar}(s)=n$. If $f\in\mathcal{F}$ and
$\operatorname{ar}(f)=n$, then an expression $f(t_1,\ldots,t_k)$ is a
well-formed term exactly when $k=n$. If $R\in\mathcal{R}$ and
$\operatorname{ar}(R)=n$, then an expression $R(t_1,\ldots,t_k)$ is a
well-formed atomic formula exactly when $k=n$.
\end{remark*}

\begin{remark*}[Interpretation]
Arity is bookkeeping for grammatical well-formedness. It determines whether a
symbol has been supplied with the correct number of arguments.
\end{remark*}

\begin{definition}[Language]
\label{def:language}
A \emph{first-order language} has two standard readings. As vocabulary, it is
the same data as a signature,
\[
  \mathcal{L}=(\mathcal{C},\mathcal{F},\mathcal{R},\operatorname{ar}).
\]
As a formal language generated by a signature $\Sigma$, it is the set of
well-formed strings built from the alphabet
\[
  A=\Sigma\cup V\cup\Lambda,
\]
where $V$ is a countable set of variables and $\Lambda$ is the fixed stock of
logical symbols and punctuation. In this second reading,
\[
  \mathcal{L}(\Sigma)
  =\{s\in A^* \mid s\in\operatorname{Terms}(\Sigma)
    \text{ or } s\in\operatorname{WFF}(\Sigma)\}.
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
The vocabulary reading records which non-logical symbols may occur. The
generated-language reading then selects from all finite strings over $A$ only
the terms and well-formed formulas admitted by the formation rules determined
by $\Sigma$.
\end{remark*}

\begin{remark*}[Interpretation]
The signature supplies the subject-specific vocabulary. The language adds the
logical grammar needed to form expressions from that vocabulary.
\end{remark*}

\begin{definition}[Constant Symbol]
\label{def:signature-constant-symbol}
A \emph{constant symbol} is a non-logical symbol that behaves syntactically as
a $0$-ary function symbol. In the streamlined convention where constants are
included among function symbols, this is the condition
\[
  \operatorname{Constant}(c)
  \iff c\in\mathcal{F}\text{ and }\operatorname{ar}(c)=0.
\]
Because its arity is $0$, the symbol $c$ is already a well-formed term and
takes no input terms.
\end{definition}

\begin{remark*}[Standard quantified statement]
If $c\in\mathcal{F}$ and $\operatorname{ar}(c)=0$, then
$c\in\operatorname{Terms}(\mathcal{L})$. In a structure
$\mathfrak{M}=(M,\mathcal{I})$, the interpretation of a constant is a unique
element of the domain:
\[
  \forall c\in\mathcal{C},\ \exists! e\in M,\ \mathcal{I}(c)=e.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A constant symbol is syntactically rigid: unlike a variable, it does not vary
with an assignment. In the arithmetic signature, the symbol $0$ is a piece of
formal syntax; under the standard interpretation, it denotes the number zero.
\end{remark*}

\begin{definition}[Function Symbol]
\label{def:signature-function-symbol}
A \emph{function symbol} is a non-logical symbol $f\in\mathcal{F}$ whose
assigned arity determines how many terms it must take as inputs. If
$\operatorname{ar}(f)=n$, then $f$ is an $n$-ary term constructor: it takes
exactly $n$ terms and produces a new term, not a truth value.
\end{definition}

\begin{remark*}[Standard quantified statement]
For every $f\in\mathcal{F}$, if $\operatorname{ar}(f)=n$, then
\[
  f(t_1,\ldots,t_k)\in\operatorname{Terms}(\mathcal{L})
  \iff k=n
\]
whenever $t_1,\ldots,t_k$ are terms. In a structure
$\mathfrak{M}=(M,\mathcal{I})$, the interpretation of $f$ is a total operation
closed on the domain:
\[
  \mathcal{I}(f):M^n\to M.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A function symbol is operation-forming syntax. In the arithmetic signature,
$S$ is unary, while $+$ and $\cdot$ are binary. Under an interpretation, each
becomes an actual operation on the chosen domain.
\end{remark*}

\begin{definition}[Relation Symbol]
\label{def:relation-symbol}
A \emph{relation symbol}, or predicate symbol, is a non-logical symbol
$R\in\mathcal{R}$ whose assigned arity determines how many terms it must take
as inputs. If $\operatorname{ar}(R)=n$, then $R$ is an $n$-ary formula
constructor: it takes exactly $n$ terms and produces an atomic formula, hence
something with a truth value rather than an object.
\end{definition}

\begin{remark*}[Standard quantified statement]
For every $R\in\mathcal{R}$, if $\operatorname{ar}(R)=n$, then
\[
  R(t_1,\ldots,t_k)\in\operatorname{WFF}(\mathcal{L})
  \iff k=n
\]
whenever $t_1,\ldots,t_k$ are terms. In a structure
$\mathfrak{M}=(M,\mathcal{I})$, the interpretation of $R$ is a subset of
$n$-tuples from the domain:
\[
  \mathcal{I}(R)\subseteq M^n.
\]
Thus $R(t_1,\ldots,t_n)$ is true exactly when the tuple of interpreted terms
lies in $\mathcal{I}(R)$.
\end{remark*}

\begin{remark*}[Interpretation]
A relation symbol is a statement-forming symbol once it is applied to the
right number of terms. In the arithmetic signature, $+$ is a binary function
symbol because it returns a number, while $<$ is a binary relation symbol
because it returns a truth value. Under the standard interpretation,
$<$ denotes the set of ordered pairs $(a,b)$ such that $a$ is less than $b$.
\end{remark*}

\begin{remark*}[Structural remarks]
The same signature may support different choices of axioms later in the
chapter. For example, a single arithmetic vocabulary can be used with stronger
or weaker axiom sets.

The same signature may also be read over different mathematical universes once
interpretations are introduced. At this stage, the signature itself records
only the common vocabulary; it does not choose any particular interpretation.
\end{remark*}
