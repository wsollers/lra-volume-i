\begin{toolkitbox}{Functions --- Quick Reference}
\small
\begin{tabular}{l l l}
\toprule
\textbf{Concept} & \textbf{Meaning} & \textbf{Detail} \\
\midrule
Function         & Left-total, right-unique relation                            & \hyperref[def:function]{$\downarrow$ Def} \\
Domain / Codomain & $\dom(f) = A$,\; $\cod(f) = B$ for $f : A \to B$            & \hyperref[def:domain-codomain]{$\downarrow$ Def} \\
Image of function & $\im(f) = \{f(a) \mid a \in A\}$                            & \hyperref[def:image-function]{$\downarrow$ Def} \\
Image of set      & $f(S) = \{f(a) \mid a \in S\}$ for $S \subseteq A$          & \hyperref[def:image-set]{$\downarrow$ Def} \\
Preimage          & $f^{-1}(T) = \{a \in A \mid f(a) \in T\}$ for $T \subseteq B$ & \hyperref[def:preimage]{$\downarrow$ Def} \\
Fiber             & $f^{-1}(\{b\})$: preimage of a singleton                    & \hyperref[def:fiber]{$\downarrow$ Def} \\
Graph             & $\{(a,b) \in A\times B \mid b = f(a)\}$                     & \hyperref[def:graph]{$\downarrow$ Def} \\
Injective         & Distinct inputs $\Rightarrow$ distinct outputs               & \hyperref[def:injective]{$\downarrow$ Def} \\
Surjective        & Every codomain element is achieved                          & \hyperref[def:surjective]{$\downarrow$ Def} \\
Bijective         & Injective and surjective                                    & \hyperref[def:bijective]{$\downarrow$ Def} \\
Identity          & $\id_A(a) = a$                                              & \hyperref[def:identity]{$\downarrow$ Def} \\
Inclusion map     & $\iota : A \hookrightarrow B$, $\iota(a) = a$ for $A \subseteq B$ & \hyperref[def:inclusion]{$\downarrow$ Def} \\
Composition       & $(g \circ f)(a) = g(f(a))$                                  & \hyperref[def:composition]{$\downarrow$ Def} \\
Inverse           & Defined for bijections; $f^{-1} \circ f = \id_A$            & \hyperref[def:inverse]{$\downarrow$ Def} \\
Left / Right inverse & Section and retraction                                  & \hyperref[def:left-right-inverse]{$\downarrow$ Def} \\
Restriction       & $f|_C : C \to B$                                            & \hyperref[def:restriction]{$\downarrow$ Def} \\
Extension         & $g : A' \to B$ with $g|_A = f$                              & \hyperref[def:extension]{$\downarrow$ Def} \\
\bottomrule
\end{tabular}
\end{toolkitbox}
