\begin{toolkitbox}{Relational Properties --- Quick Reference}
\small
\begin{tabular}{l l l l}
\toprule
\textbf{Property} & \textbf{Formal definition} & \textbf{Canonical example} & \textbf{Detail} \\
\midrule
Reflexive      & $\forall a,\; (a,a)\in R$                           & $=$ on any set      & \hyperref[def:reflexive]{$\downarrow$ Def} \\
Irreflexive    & $\forall a,\; (a,a)\notin R$                        & $<$ on $\mathbb{N}$ & \hyperref[def:irreflexive]{$\downarrow$ Def} \\
Symmetric      & $(a,b)\in R \Rightarrow (b,a)\in R$                 & ``same age as''     & \hyperref[def:symmetric]{$\downarrow$ Def} \\
Antisymmetric  & $(a,b),(b,a)\in R \Rightarrow a=b$                  & $\leq$ on $\mathbb{N}$ & \hyperref[def:antisymmetric]{$\downarrow$ Def} \\
Asymmetric     & $(a,b)\in R \Rightarrow (b,a)\notin R$              & $<$ on $\mathbb{N}$ & \hyperref[def:asymmetric]{$\downarrow$ Def} \\
Transitive     & $(a,b),(b,c)\in R \Rightarrow (a,c)\in R$           & $\leq$ on $\mathbb{N}$ & \hyperref[def:transitive]{$\downarrow$ Def} \\
Total (Connex) & $(a,b)\in R \lor (b,a)\in R$                        & $\leq$ on $\mathbb{R}$ & \hyperref[def:total-rel]{$\downarrow$ Def} \\
\midrule
\multicolumn{4}{l}{\textit{Structural classes formed by combining properties:}} \\
\midrule
Equivalence    & Reflexive + Symmetric + Transitive                  & $=$ on any set      & \hyperref[def:equivalence-rel]{$\downarrow$ Def} \\
Preorder       & Reflexive + Transitive                              & $\leq$ on $\mathbb{N}$ & \hyperref[def:preorder]{$\downarrow$ Def} \\
Partial order  & Reflexive + Antisymmetric + Transitive              & $\subseteq$ on $\mathcal{P}(A)$ & \hyperref[def:partial-order]{$\downarrow$ Def} \\
Total order    & Partial order + Total                               & $\leq$ on $\mathbb{R}$ & \hyperref[def:total-order]{$\downarrow$ Def} \\
\bottomrule
\end{tabular}
\end{toolkitbox}
