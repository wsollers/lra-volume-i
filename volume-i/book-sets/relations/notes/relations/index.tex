\begin{toolkitbox}{Relations --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[def:ordered-pair]{Ordered pair}      & $(a,b) := \{\{a\},\{a,b\}\}$; identity iff both coords equal  \\
\hyperref[def:cartesian-product-rel]{Cartesian product}  & $A \times B$: all ordered pairs from $A$ and $B$             \\
\hyperref[def:relation]{Relation}          & Any subset $R \subseteq A \times B$                           \\
\hyperref[def:relation-domain]{Domain of $R$}     & Inputs that occur in $R$                                      \\
\hyperref[def:relation-range]{Range of $R$}      & Outputs that occur in $R$                                     \\
\hyperref[def:converse-relation]{Converse}          & Reverse every ordered pair                                    \\
\hyperref[def:relation-identity]{Identity relation}  & Diagonal relation $\Delta_A$                                 \\
\hyperref[def:relation-composition]{Relation composition}  & Compose through an intermediate witness                  \\
\bottomrule
\end{tabular}
\end{toolkitbox}

\begin{toolkitbox}{Relation Algebra and Classification --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[def:null-relation]{Null relation}        & $\varnothing$                                               \\
\hyperref[def:universal-relation]{Universal relation}   & $A\times B$                                                 \\
\hyperref[def:relation-inclusion]{Relation inclusion}   & $R\subseteq S$ as set inclusion                             \\
\hyperref[def:relation-equality]{Relation equality}    & Mutual relation inclusion                                   \\
\hyperref[def:relation-union]{Union of relations}   & Pair belongs to at least one relation                       \\
\hyperref[def:relation-intersection]{Intersection}         & Pair belongs to both relations                              \\
\hyperref[def:relation-difference]{Difference}           & Pair belongs to first relation but not second               \\
\hyperref[def:relation-complement]{Complement}           & Pairs in $A\times B$ not in $R$                             \\
\hyperref[thm:relation-converse-laws]{Converse laws}        & Converse reverses composition and preserves Boolean operations \\
\hyperref[thm:relation-composition-boolean]{Composition laws}     & Associative; distributes over unions, not all Boolean operations \\
\hyperref[def:one-to-many-relation]{One-to-many}          & One input has two distinct outputs                          \\
\hyperref[def:many-to-one-relation]{Many-to-one}          & Two distinct inputs share one output                        \\
\hyperref[def:many-to-many-relation]{Many-to-many}         & One-to-many and many-to-one phenomena both occur            \\
\hyperref[def:many-place-relation]{Ternary relation}     & Subset of $A\times B\times C$                               \\
\bottomrule
\end{tabular}
\end{toolkitbox}

\begin{toolkitbox}{Relational Properties --- Quick Reference}
\small
\begin{tabular}{l p{0.68\linewidth}}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[def:reflexive]{Reflexive}       & $\forall a,\; (a,a)\in R$; example: $=$ on any set \\
\hyperref[def:irreflexive]{Irreflexive}     & $\forall a,\; (a,a)\notin R$; example: $<$ on $\mathbb{N}$ \\
\hyperref[def:symmetric]{Symmetric}       & $(a,b)\in R \Rightarrow (b,a)\in R$; example: ``same age as'' \\
\hyperref[def:antisymmetric]{Antisymmetric}   & $(a,b),(b,a)\in R \Rightarrow a=b$; example: $\leq$ on $\mathbb{N}$ \\
\hyperref[def:asymmetric]{Asymmetric}      & $(a,b)\in R \Rightarrow (b,a)\notin R$; example: $<$ on $\mathbb{N}$ \\
\hyperref[def:transitive]{Transitive}      & $(a,b),(b,c)\in R \Rightarrow (a,c)\in R$; example: $\leq$ on $\mathbb{N}$ \\
\hyperref[def:total-rel]{Total (Connex)}  & $(a,b)\in R \lor (b,a)\in R$; example: $\leq$ on $\mathbb{R}$ \\
\midrule
\multicolumn{2}{l}{\textit{Structural classes formed by combining properties:}} \\
\midrule
\hyperref[def:equivalence-rel]{Equivalence}     & Reflexive + symmetric + transitive; example: $=$ on any set \\
\hyperref[def:preorder]{Preorder}        & Reflexive + transitive; example: $\leq$ on $\mathbb{N}$ \\
\hyperref[def:partial-order]{Partial order}   & Reflexive + antisymmetric + transitive; example: $\subseteq$ on $\mathcal{P}(A)$ \\
\hyperref[def:strict-partial-order-relations]{Strict partial order}   & Irreflexive + transitive, hence asymmetric; example: $<$ on $\mathbb{R}$ \\
\hyperref[def:total-order]{Total order}     & Partial order + totality; example: $\leq$ on $\mathbb{R}$ \\
\bottomrule
\end{tabular}
\end{toolkitbox}
