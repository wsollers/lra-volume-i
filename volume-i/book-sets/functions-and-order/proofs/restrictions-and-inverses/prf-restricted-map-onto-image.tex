\begin{proposition*}[Restricted Map Is Onto Its Restricted Image]
Let $f:A\to B$ and let $C\subseteq A$. Then $f|_C:C\to f(C)$ is surjective.
\end{proposition*}

\begin{proof}[Professional Standard Proof]
\LRAProofBodyStart
Let $y\in f(C)$. Then $y=f(c)$ for some $c\in C$. Since
$f|_C(c)=f(c)$, the restricted map reaches $y$.
\end{proof}

\begin{proof}[Detailed Learning Proof]
\LRAProofBodyStart
The codomain $f(C)$ contains exactly the values produced by inputs from $C$.
Therefore every element of that codomain is reached by the restricted function.
\end{proof}
