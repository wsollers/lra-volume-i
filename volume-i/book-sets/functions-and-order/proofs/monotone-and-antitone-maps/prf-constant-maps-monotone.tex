\begin{proposition*}[Constant Maps Are Monotone]
Every constant function between ordered sets is order-preserving.
\end{proposition*}

\begin{proof}[Professional Standard Proof]
\LRAProofBodyStart
Let $f:M\to M'$ be constant with value $c$. If $x\leq y$ in $M$, then
$f(x)=c=f(y)$, and hence $f(x)\leq' f(y)$ by reflexivity.
\end{proof}

\begin{proof}[Detailed Learning Proof]
\LRAProofBodyStart
Order-preservation asks that comparable inputs produce comparable outputs in
the same direction. A constant map produces equal outputs, and equality gives
both order directions in a partial order.
\end{proof}
