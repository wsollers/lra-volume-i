\begin{proposition*}[Order Embeddings Are Injective, Revisited]
Every order embedding between partial orders is injective.
\end{proposition*}

\begin{proof}[Professional Standard Proof]
\LRAProofBodyStart
Let $f:A\to B$ be an order embedding and suppose $f(x)=f(y)$. Then
$f(x)\leq' f(y)$ and $f(y)\leq' f(x)$. Since $f$ reflects order,
$x\leq y$ and $y\leq x$. Antisymmetry gives $x=y$.
\end{proof}

\begin{proof}[Detailed Learning Proof]
\LRAProofBodyStart
Equality of outputs gives both order inequalities between those outputs.
Order-reflection pulls both inequalities back to the domain. In a partial
order, two elements ordered both ways are equal.
\end{proof}
