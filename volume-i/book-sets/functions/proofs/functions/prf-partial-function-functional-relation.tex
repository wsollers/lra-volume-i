\begin{proposition*}[Partial Functions Are Functional Relations]
For sets $A$ and $B$ and a relation $R\subseteq A\times B$, $R$ is a partial
function from $A$ to $B$ if and only if $R$ is functional from $A$ to $B$.
\end{proposition*}
