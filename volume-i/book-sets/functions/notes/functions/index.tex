\begin{toolkitbox}{Functions --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[def:functional-relation]{Functional relation}  & Each input has at most one output                         \\
\hyperref[def:total-relation-on-domain]{Total relation on $A$}  & Each input has at least one output                       \\
\hyperref[def:function]{Function}          & Left-total, right-unique relation                            \\
\hyperref[def:partial-function]{Partial function}  & Functional, not necessarily total                            \\
\hyperref[def:function-equality]{Function equality}  & Same domain, codomain, and pointwise values                 \\
\hyperref[def:many-place-functional-relation]{Many-place function}  & Functional relation in the last coordinate                 \\
\hyperref[def:domain]{Domain}            & $\dom(f)=A$ for $f:A\to B$                                 \\
\hyperref[def:codomain]{Codomain}          & $\cod(f)=B$ for $f:A\to B$                                  \\
\hyperref[def:image-function]{Image of function}  & $\im(f) = \{f(a) \mid a \in A\}$                            \\
\hyperref[def:image-set]{Image of set}       & $f(S) = \{f(a) \mid a \in S\}$ for $S \subseteq A$          \\
\hyperref[def:preimage]{Preimage}           & $f^{-1}(T) = \{a \in A \mid f(a) \in T\}$ for $T \subseteq B$ \\
\hyperref[def:fiber]{Fiber}              & $f^{-1}(\{b\})$: preimage of a singleton                    \\
\hyperref[def:graph]{Graph}              & $\{(a,b) \in A\times B \mid b = f(a)\}$                     \\
\hyperref[def:injective]{Injective}          & Distinct inputs $\Rightarrow$ distinct outputs               \\
\hyperref[def:surjective]{Surjective}         & Every codomain element is achieved                          \\
\hyperref[def:bijective]{Bijective}          & Injective and surjective                                    \\
\midrule
\hyperref[thm:function-total-functional-characterization]{Function test}      & Function iff total and functional                           \\
\hyperref[prop:one-to-one-converse-functional]{One-one test}       & Injective iff converse graph is functional                  \\
\hyperref[prop:bijection-unique-both-directions]{Bijective test}     & Unique correspondence in both directions                    \\
\midrule
\hyperref[def:identity]{Identity}           & $\id_A(a) = a$                                              \\
\hyperref[def:inclusion]{Inclusion map}      & $\iota : A \hookrightarrow B$, $\iota(a) = a$ for $A \subseteq B$ \\
\hyperref[def:composition]{Composition}        & $(g \circ f)(a) = g(f(a))$                                  \\
\hyperref[def:invertible-function]{Invertible}       & Some $g:B\to A$ satisfies $g\circ f=\id_A$ and $f\circ g=\id_B$ \\
\hyperref[def:inverse]{Inverse}            & Defined for bijections; $f^{-1} \circ f = \id_A$            \\
\hyperref[def:left-inverse-function]{Left inverse}       & $g\circ f=\id_A$                                           \\
\hyperref[def:right-inverse-function]{Right inverse}      & $f\circ h=\id_B$                                           \\
\hyperref[def:restriction]{Restriction}        & $f|_C : C \to B$                                            \\
\hyperref[def:fn-extension]{Extension}          & $g : A' \to B$ with $g|_A = f$                              \\
\hyperref[def:empty-function]{Empty function}     & Unique function $\varnothing\to B$                          \\
\bottomrule
\end{tabular}
\end{toolkitbox}

\begin{toolkitbox}{Composition \& Inverses --- Quick Reference}
\small
\begin{tabular}{l p{0.68\linewidth}}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[thm:comp-assoc]{Associativity of $\circ$}  & $h \circ (g \circ f) = (h \circ g) \circ f$; proved element-wise \\
\hyperref[thm:comp-id]{Identity and $\circ$}      & $f \circ \id_A = \id_B \circ f = f$; proved element-wise \\
\hyperref[thm:comp-inj-surj]{Inj/Surj under $\circ$}    & Closed under composition, with partial converses; proved element-wise \\
\hyperref[thm:inverse-char]{Inverse characterization}  & $f^{-1}\circ f = \id_A$ and $f \circ f^{-1} = \id_B$ characterize bijectivity \\
\hyperref[thm:inverse-comp]{Inverse of composition}    & $(g \circ f)^{-1} = f^{-1} \circ g^{-1}$ by verifying both sides \\
\hyperref[thm:one-sided]{One-sided inverses}        & Left inverse iff injective; right inverse iff surjective \\
\hyperref[thm:preimage-ops]{Preimage preserves ops}    & $f^{-1}$ commutes with $\cup,\cap,\setminus,c$ by element-wise proof \\
\hyperref[thm:indexed-preimage-ops]{Indexed preimage ops} & $f^{-1}$ commutes with indexed $\cup$ and $\cap$ \\
\hyperref[thm:image-ops]{Image and set ops}         & Images preserve $\cup$ and contain $\subseteq$ for $\cap$; counterexamples mark failures \\
\hyperref[thm:image-preimage-adjunction]{Image/preimage adjunction}  & $S\subseteq f^{-1}(f(S))$ and $f(f^{-1}(T))\subseteq T$ \\
\hyperref[prop:preimage-under-inverse-function]{Preimage under inverse} & $(f^{-1})^{-1}(S)=f(S)$ for bijective $f$ \\
\bottomrule
\end{tabular}
\end{toolkitbox}
