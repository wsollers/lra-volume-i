\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[def:same-cardinality]{Same cardinality}      & Bijection $f : A \to B$ exists; written $A \sim B$            \\
\hyperref[def:finite-infinite]{Finite set}            & $A \sim \{1,\dots,n\}$ for some $n \in \mathbb{N}$, or $A = \varnothing$ \\
\hyperref[def:countable]{Countably infinite}    & $A \sim \mathbb{N}$                                           \\
\hyperref[def:countable]{Countable}             & Finite or countably infinite                                  \\
\hyperref[def:countable]{Uncountable}           & Infinite but not countable                                    \\
\hyperref[def:cardinality-order]{$|A| \leq |B|$}        & Injection $A \hookrightarrow B$ exists                        \\
\hyperref[thm:Q-countable]{$\mathbb{Q}$ countable}  & Diagonal enumeration                                       \\
\hyperref[thm:R-uncountable]{$\mathbb{R}$ uncountable}  & Cantor diagonal argument                                \\
\hyperref[thm:schroder-bernstein]{Schr\"{o}der--Bernstein}   & $|A|\leq|B|$ and $|B|\leq|A|$ $\Rightarrow$ $A \sim B$      \\
\hyperref[thm:cantor]{Cantor's theorem}      & $|A| < |\mathcal{P}(A)|$ for every set $A$                   \\
\hyperref[thm:countable-union]{Countable union}       & Countable union of countable sets is countable                \\
\bottomrule
\end{tabular}
