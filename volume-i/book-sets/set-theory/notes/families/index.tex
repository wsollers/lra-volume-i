\begin{toolkitbox}{Indexed Families --- Quick Reference}
\small
\begin{tabular}{l l l}
\toprule
\textbf{Concept} & \textbf{Meaning} & \textbf{Detail} \\
\midrule
Indexed family      & Function $F : I \to \mathcal{P}(U)$; written $\{A_i\}_{i \in I}$ & \hyperref[def:indexed-family]{$\downarrow$ Def} \\
Indexed union       & $\bigcup_{i \in I} A_i$: in at least one $A_i$                    & \hyperref[def:indexed-union]{$\downarrow$ Def} \\
Indexed intersection & $\bigcap_{i \in I} A_i$: in every $A_i$                         & \hyperref[def:indexed-intersection]{$\downarrow$ Def} \\
Pairwise disjoint   & $i \neq j \Rightarrow A_i \cap A_j = \varnothing$                  & \hyperref[def:pairwise-disjoint]{$\downarrow$ Def} \\
Cover               & $\bigcup_{C \in \mathcal{C}} C = A$                               & \hyperref[def:set-cover]{$\downarrow$ Def} \\
Indexed De Morgan   & Complements exchange indexed unions and intersections             & \hyperref[thm:indexed-de-morgan]{$\downarrow$ Thm} \\
Arbitrary product   & $\prod_{i \in I} A_i$: choice functions $f : I \to \bigcup A_i$   & \hyperref[def:arbitrary-product]{$\downarrow$ Def} \\
\bottomrule
\end{tabular}
\end{toolkitbox}

\begin{toolkitbox}{Covers and FIP --- Quick Reference}
\small
\begin{tabular}{l l l}
\toprule
\textbf{Concept} & \textbf{Meaning} & \textbf{Detail} \\
\midrule
Cover of $A$     & $\bigcup_{C \in \mathcal{C}} C \supseteq A$                     & \hyperref[def:cover-full]{$\downarrow$ Def} \\
Subcover         & Sub-collection of a cover that is still a cover                 & \hyperref[def:subcover]{$\downarrow$ Def} \\
Finite cover     & Cover with finitely many members                                & \hyperref[def:finite-cover]{$\downarrow$ Def} \\
Finite subcover  & Finite subcollection that is still a cover                      & \hyperref[def:subcover]{$\downarrow$ Def} \\
Open cover       & Cover whose members are all open sets                           & \hyperref[def:open-cover]{$\downarrow$ Def} \\
Finite Intersection Property & Every finite subcollection has nonempty intersection & \hyperref[def:fip]{$\downarrow$ Def} \\
FIP--cover duality & $\mathcal{C}$ covers $A$ $\iff$ closed complements fail FIP   & \hyperref[prop:fip-duality]{$\downarrow$ Prop} \\
\bottomrule
\end{tabular}
\end{toolkitbox}
