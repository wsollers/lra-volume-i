\begin{toolkitbox}{Indexed Families --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Concept}  & \textbf{Meaning} \\
\midrule
\hyperref[def:indexed-family]{Indexed family}       & Function $F : I \to \mathcal{P}(U)$; written $\{A_i\}_{i \in I}$ \\
\hyperref[def:indexed-union]{Indexed union}        & $\bigcup_{i \in I} A_i$: in at least one $A_i$                    \\
\hyperref[def:indexed-intersection]{Indexed intersection}  & $\bigcap_{i \in I} A_i$: in every $A_i$                         \\
\hyperref[def:pairwise-disjoint]{Pairwise disjoint}    & $i \neq j \Rightarrow A_i \cap A_j = \varnothing$                  \\
\hyperref[def:set-cover]{Cover}                & $\bigcup_{C \in \mathcal{C}} C = A$                               \\
\hyperref[thm:indexed-de-morgan]{Indexed De Morgan}    & Complements exchange indexed unions and intersections             \\
\hyperref[thm:indexed-distributivity]{Indexed distributivity} & $\cap$ distributes over indexed $\cup$; dually for $\cup$ over $\cap$ \\
\hyperref[def:arbitrary-product]{Arbitrary product}    & $\prod_{i \in I} A_i$: choice functions $f : I \to \bigcup A_i$   \\
\bottomrule
\end{tabular}
\end{toolkitbox}

\begin{toolkitbox}{Covers and FIP --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Concept}  & \textbf{Meaning} \\
\midrule
\hyperref[def:cover-full]{Cover of $A$}      & $\bigcup_{C \in \mathcal{C}} C \supseteq A$                     \\
\hyperref[def:subcover]{Subcover}          & Sub-collection of a cover that is still a cover                 \\
\hyperref[def:finite-cover]{Finite cover}      & Cover with finitely many members                                \\
\hyperref[def:subcover]{Finite subcover}   & Finite subcollection that is still a cover                      \\
\hyperref[def:open-cover]{Open cover}        & Cover whose members are all open sets                           \\
\hyperref[def:fip]{Finite Intersection Property}  & Every finite subcollection has nonempty intersection \\
\hyperref[prop:fip-duality]{FIP--cover duality}  & $\mathcal{C}$ covers $A$ $\iff$ closed complements fail FIP   \\
\bottomrule
\end{tabular}
\end{toolkitbox}
