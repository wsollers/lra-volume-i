\begin{toolkitbox}{Set Operations --- Quick Reference}
\small
\begin{tabular}{l l l}
\toprule
\textbf{Concept}  & \textbf{Notation}  & \textbf{Membership condition} \\
\midrule
\hyperref[def:empty-set]{Empty set}         & $\varnothing$              & $x \in \varnothing$ never         \\
\hyperref[def:subset]{Subset}            & $A \subseteq B$            & $x \in A \Rightarrow x \in B$     \\
\hyperref[def:proper-subset]{Proper subset}     & $A \subsetneq B$           & $A \subseteq B$ and $A \neq B$    \\
\hyperref[def:set-equality]{Set equality}      & $A = B$                    & same elements both ways            \\
\hyperref[def:union]{Union}             & $A \cup B$                 & in $A$ or in $B$                  \\
\hyperref[def:intersection]{Intersection}      & $A \cap B$                 & in $A$ and in $B$                 \\
\hyperref[def:set-difference]{Set difference}    & $A \setminus B$            & in $A$ but not $B$                \\
\hyperref[def:sym-diff]{Symmetric diff.}   & $A \triangle B$            & in exactly one of $A$, $B$        \\
\hyperref[def:complement]{Complement}        & $A^c = U \setminus A$      & in $U$ but not $A$                \\
\hyperref[def:cartesian-product]{Cartesian product} & $A \times B$               & all ordered pairs $(a,b)$         \\
\hyperref[def:power-set]{Power set}         & $\mathcal{P}(A)$           & all subsets of $A$                \\
\hyperref[thm:union-monotone-inclusion]{Union monotone} & $A\subseteq B$              & $A\cup C\subseteq B\cup C$ \\
\hyperref[thm:intersection-monotone-inclusion]{Intersection monotone} & $A\subseteq B$              & $A\cap C\subseteq B\cap C$ \\
\hyperref[thm:power-set-monotone-inclusion]{Power set monotone} & $A\subseteq B$              & $\mathcal{P}(A)\subseteq\mathcal{P}(B)$ \\
\hyperref[thm:complement-antitone-inclusion]{Complement antitone} & $A\subseteq B$              & $B^c\subseteq A^c$ \\
\hyperref[thm:set-difference-monotone-left]{Difference left monotone} & $A\subseteq B$              & $A\setminus C\subseteq B\setminus C$ \\
\hyperref[thm:set-difference-antitone-right]{Difference right antitone} & $C\subseteq D$              & $A\setminus D\subseteq A\setminus C$ \\
\hyperref[thm:de-morgan]{De Morgan}         & $(A \cup B)^c = A^c \cap B^c$  & complement of union        \\
\bottomrule
\end{tabular}
\end{toolkitbox}

\begin{toolkitbox}{Set Algebra Laws --- Quick Reference}
\small
\begin{tabular}{l l l}
\toprule
\textbf{Law}  & \textbf{Union form}  & \textbf{Intersection form} \\
\midrule
\hyperref[thm:commutativity]{Commutativity}   & $A \cup B = B \cup A$           & $A \cap B = B \cap A$          \\
\hyperref[thm:associativity]{Associativity}   & $(A \cup B) \cup C = A \cup (B \cup C)$   & analogous   \\
\hyperref[thm:distributivity]{Distributivity}  & $A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$  & dual   \\
\hyperref[thm:identity-absorption]{Identity}        & $A \cup \varnothing = A$         & $A \cap U = A$                 \\
\hyperref[thm:identity-absorption]{Absorption}      & $A \cup (A \cap B) = A$          & $A \cap (A \cup B) = A$        \\
\hyperref[thm:involution]{Involution}      & $(A^c)^c = A$                   & ---                            \\
\bottomrule
\end{tabular}
\end{toolkitbox}
