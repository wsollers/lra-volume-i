\begin{toolkitbox}{Set Operations --- Quick Reference}
\small
\begin{tabular}{l p{0.68\linewidth}}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[def:empty-set]{Empty set}         & $\varnothing$; $x \in \varnothing$ never \\
\hyperref[def:subset]{Subset}            & $A \subseteq B$ means $x \in A \Rightarrow x \in B$ \\
\hyperref[def:proper-subset]{Proper subset}     & $A \subsetneq B$ means $A \subseteq B$ and $A \neq B$ \\
\hyperref[def:set-equality]{Set equality}      & $A = B$ means the two sets have the same elements both ways \\
\hyperref[def:union]{Union}             & $A \cup B$ contains elements in $A$ or in $B$ \\
\hyperref[def:intersection]{Intersection}      & $A \cap B$ contains elements in $A$ and in $B$ \\
\hyperref[def:set-difference]{Set difference}    & $A \setminus B$ contains elements in $A$ but not in $B$ \\
\hyperref[def:sym-diff]{Symmetric diff.}   & $A \triangle B$ contains elements in exactly one of $A$, $B$ \\
\hyperref[def:complement]{Complement}        & $A^c = U \setminus A$ contains elements in $U$ but not in $A$ \\
\hyperref[def:cartesian-product]{Cartesian product} & $A \times B$ is the set of ordered pairs $(a,b)$ \\
\hyperref[def:power-set]{Power set}         & $\mathcal{P}(A)$ is the set of all subsets of $A$ \\
\hyperref[thm:union-monotone-inclusion]{Union monotone} & From $A\subseteq B$ infer $A\cup C\subseteq B\cup C$ \\
\hyperref[thm:intersection-monotone-inclusion]{Intersection monotone} & From $A\subseteq B$ infer $A\cap C\subseteq B\cap C$ \\
\hyperref[thm:power-set-monotone-inclusion]{Power set monotone} & From $A\subseteq B$ infer $\mathcal{P}(A)\subseteq\mathcal{P}(B)$ \\
\hyperref[thm:complement-antitone-inclusion]{Complement antitone} & From $A\subseteq B$ infer $B^c\subseteq A^c$ \\
\hyperref[thm:set-difference-monotone-left]{Difference left monotone} & From $A\subseteq B$ infer $A\setminus C\subseteq B\setminus C$ \\
\hyperref[thm:set-difference-antitone-right]{Difference right antitone} & From $C\subseteq D$ infer $A\setminus D\subseteq A\setminus C$ \\
\hyperref[thm:de-morgan]{De Morgan}         & $(A \cup B)^c = A^c \cap B^c$ describes the complement of a union \\
\bottomrule
\end{tabular}
\end{toolkitbox}

\begin{toolkitbox}{Set Algebra Laws --- Quick Reference}
\small
\begin{tabular}{l p{0.68\linewidth}}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[thm:commutativity]{Commutativity}   & $A \cup B = B \cup A$ and $A \cap B = B \cap A$ \\
\hyperref[thm:associativity]{Associativity}   & Unions and intersections may be regrouped without changing the set \\
\hyperref[thm:distributivity]{Distributivity}  & Intersection distributes over union, and the dual law also holds \\
\hyperref[thm:identity-absorption]{Identity}        & $A \cup \varnothing = A$ and $A \cap U = A$ \\
\hyperref[thm:identity-absorption]{Absorption}      & $A \cup (A \cap B) = A$ and $A \cap (A \cup B) = A$ \\
\hyperref[thm:involution]{Involution}      & $(A^c)^c = A$ \\
\bottomrule
\end{tabular}
\end{toolkitbox}
