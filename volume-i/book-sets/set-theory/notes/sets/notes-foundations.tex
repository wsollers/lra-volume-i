\subsection{Sets, Membership, and ZFC Axioms}

\begin{axiombox}{Axiom (Extensionality)}
\begin{axiom}[Axiom of Extensionality]
\label{ax:extensionality}
Two sets are equal iff they have the same elements.
\[
\forall A \, \forall B \,
\Bigl( A = B \iff \forall x \, (x \in A \leftrightarrow x \in B) \Bigr).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
  \forall A\,\forall B,\quad
  A=B \iff \forall x,\ (x\in A\leftrightarrow x\in B).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{SetEqual}(A,B)
  \iff
  \forall x,\ (x\in A\leftrightarrow x\in B).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Sets are identified entirely by their members.
\end{remark*}

\begin{axiombox}{Axiom (Empty Set)}
\begin{axiom}[Axiom of Empty Set]
\label{ax:empty-set}
There exists a set with no elements.
\[
\exists A \, \forall x \, (x \notin A).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
  \exists A\,\forall x,\ x\notin A.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{EmptySet}(A)
  \iff
  \forall x,\ x\notin A.
\]
\end{remark*}

\begin{remark*}[Interpretation]
There is a set with no elements.
\end{remark*}

\begin{axiombox}{Axiom (Pairing)}
\begin{axiom}[Axiom of Pairing]
\label{ax:pairing}
For any two sets, there exists a set containing exactly those two sets.
\[
\forall A \, \forall B \, \exists C \,
\forall x \, (x \in C \leftrightarrow (x = A \lor x = B)).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
  \forall A\,\forall B\,\exists C\,\forall x,\quad
  x\in C \iff (x=A\vee x=B).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{PairingSet}(C,A,B)
  \iff
  \forall x,\ (x\in C\leftrightarrow (x=A\vee x=B)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Any two sets can be collected into a set containing exactly those two objects.
\end{remark*}

\NoLocalDependencies

\begin{axiombox}{Axiom (Union)}
\begin{axiom}[Axiom of Union]
\label{ax:union}
For any family of sets, there exists a set containing exactly the elements of
those sets.
\[
\forall A \, \exists U \,
\forall x \,
\bigl( x \in U \leftrightarrow \exists B \, (B \in A \land x \in B) \bigr).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
  \forall A\,\exists U\,\forall x,\quad
  x\in U \iff \exists B,\ (B\in A\wedge x\in B).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{UnionFamilySet}(U,A)
  \iff
  \forall x,\ (x\in U\leftrightarrow \exists B,\ (B\in A\wedge x\in B)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The members of all sets in a set-indexed family can be collected into one set.
\end{remark*}

\begin{axiombox}{Axiom (Power Set)}
\begin{axiom}[Axiom of Power Set]
\label{ax:power-set}
For any set, there exists a set of all its subsets.
\[
\forall A \, \exists P \,
\forall x \, (x \in P \leftrightarrow x \subseteq A).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
  \forall A\,\exists P\,\forall x,\quad
  x\in P \iff x\subseteq A.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{PowerSet}(P,A)
  \iff
  \forall x,\ (x\in P\leftrightarrow \operatorname{Subset}(x,A)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Every set has a set whose members are exactly its subsets.
\end{remark*}

\begin{axiombox}{Axiom (Infinity)}
\begin{axiom}[Axiom of Infinity]
\label{ax:infinity}
There exists an infinite set.
\[
\exists A \,
\Bigl( \varnothing \in A \land
\forall x \, (x \in A \rightarrow x \cup \{x\} \in A) \Bigr).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
  \exists A,\quad
  \varnothing\in A\wedge
  \forall x,\ (x\in A\to x\cup\{x\}\in A).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{InductiveSet}(A)
  \iff
  \varnothing\in A\wedge
  \forall x,\ (x\in A\to x\cup\{x\}\in A).
\]
\end{remark*}

\begin{remark*}[Interpretation]
There is a set closed under the successor operation starting from the empty set.
\end{remark*}

\begin{axiombox}{Axiom Schema (Separation)}
\begin{axiom}[Axiom Schema of Separation]
\label{ax:separation}
Given a set and a property, there exists a subset containing exactly the
elements satisfying that property. For any formula $\varphi(x)$,
\[
\forall A \, \exists B \,
\forall x \, (x \in B \leftrightarrow (x \in A \land \varphi(x))).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
  \forall A\,\exists B\,\forall x,\quad
  x\in B \iff (x\in A\wedge \varphi(x)).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{SeparationSubset}(B,A,\varphi)
  \iff
  \forall x,\ (x\in B\leftrightarrow (x\in A\wedge \varphi(x))).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A definable property can carve out a subset of an already given set.
\end{remark*}

\begin{axiombox}{Axiom Schema (Replacement)}
\begin{axiom}[Axiom Schema of Replacement]
\label{ax:replacement}
If $\varphi(x,y)$ defines a functional relation on a set $A$, then the image
of $A$ under $\varphi$ is a set. For any formula $\varphi(x,y)$,
\[
\forall A\,
\Bigl(
\bigl(\forall x \in A\, \exists! y\, \varphi(x,y)\bigr)
\rightarrow
\exists B\, \forall y\,
\bigl(y \in B \leftrightarrow \exists x \in A\, \varphi(x,y)\bigr)
\Bigr).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
  \forall A,\quad
  \bigl(\forall x\in A\,\exists!y\,\varphi(x,y)\bigr)
  \to
  \exists B\,\forall y,\ (y\in B\leftrightarrow \exists x\in A\,\varphi(x,y)).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{ReplacementImage}(B,A,\varphi)
  \iff
  \forall y,\ (y\in B\leftrightarrow \exists x\in A\,\varphi(x,y)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The image of a set under a definable functional relation is again a set.
\end{remark*}

\begin{axiombox}{Axiom (Foundation)}
\begin{axiom}[Axiom of Foundation]
\label{ax:foundation}
Every nonempty set has an $\in$-minimal element.
\[
\forall A \,
\Bigl( A \neq \varnothing \rightarrow
\exists x \in A \, (x \cap A = \varnothing) \Bigr).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
  \forall A,\quad
  A\neq\varnothing \to \exists x\in A,\ x\cap A=\varnothing.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{FoundationMinimalElement}(x,A)
  \iff
  x\in A\wedge x\cap A=\varnothing.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Every nonempty set has an element disjoint from it, preventing infinite
descending membership chains.
\end{remark*}

\begin{axiombox}{Axiom (Choice)}
\begin{axiom}[Axiom of Choice]
\label{ax:choice}
For any family of nonempty sets, there exists a selection function.
\[
\forall A \,
\Bigl(
(\forall B \in A \, (B \neq \varnothing))
\rightarrow
\exists f \,
\forall B \in A \, (f(B) \in B)
\Bigr).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
  \forall A,\quad
  \bigl(\forall B\in A,\ B\neq\varnothing\bigr)
  \to
  \exists f\,\forall B\in A,\ f(B)\in B.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{ChoiceFunction}(f,A)
  \iff
  \forall B\in A,\ f(B)\in B.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Every family of nonempty sets has a function choosing one element from each
member of the family.
\end{remark*}

\begin{definitionbox}{Definition (Set and Membership)}
\begin{definition}[Set and Membership]\label{def:set-membership}
In axiomatic set theory, the notions of \emph{set} and \emph{membership} are
\emph{primitive}.

\begin{itemize}
  \item A \emph{set} is an object.
  \item \emph{Membership} is a binary relation, denoted by $\in$, between objects.
\end{itemize}

If $x$ is an object and $A$ is a set, the statement $x \in A$ is read as
``$x$ is an element of $A$.'' No definition of ``set'' or ``$\in$'' is given
in more basic terms. Their meaning is determined entirely by the axioms
governing them.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
The expression $x\in A$ is a primitive membership assertion.
\end{remark*}

\begin{remark*}[Interpretation]
Set theory begins with objects called sets and a primitive relation expressing
membership of one object in another.
\end{remark*}

\begin{remark*}[Exposition]
Primitive means that set and membership are not defined in terms of simpler
concepts. Their meaning is fixed implicitly by stating the axioms they must
obey. This is the standard approach in formal mathematics: rules of behavior replace
informal definitions.
\end{remark*}

\begin{remark*}[Exposition]
All subsequent notions---subsets, ordered pairs, relations, functions,
number systems---are \emph{definitions} introduced within this axiomatic
framework. Every proved result is a \emph{theorem} derived logically from the
axioms via the rules of inference.
\end{remark*}

\begin{remark*}[Exposition]
The axioms are not statements to be proved, but rules specifying how $\in$
behaves and which sets are permitted to exist. From this point onward, set
theory functions as the underlying language of mathematics: all reasoning
about mathematical objects is ultimately grounded in these axioms, even when
they are not cited explicitly.
\end{remark*}

\begin{remark*}[Exposition]
Separation is a schema: one axiom instance per definable property $\varphi$.
It carves out subsets of an already-existing set. Replacement is strictly
stronger: it can produce sets whose elements are not already contained in any
pre-existing set, allowing the construction of large stages of the cumulative
hierarchy.
\end{remark*}

\begin{remark*}[Exposition]
The Axiom of Choice is equivalent to many statements used later in analysis,
including Zorn's Lemma, the well-ordering theorem, and the statement that every
surjective function has a right inverse. It is independent of the other ZFC
axioms.
\end{remark*}
