\begin{theorem*}[De Morgan's Laws]
Let $U$ be a universe and let $A,B \subseteq U$. Then
\[
(A \cup B)^c = A^c \cap B^c
\quad\text{and}\quad
(A \cap B)^c = A^c \cup B^c.
\]
\end{theorem*}
