\begin{theorem*}[Intersection is Inclusion-Monotone]
If $A\subseteq B$, then $A\cap C\subseteq B\cap C$ and
$C\cap A\subseteq C\cap B$.
\end{theorem*}

\begin{proof}[Professional Standard Proof]
\LRAProofBodyStart
Assume $A\subseteq B$. Let $x\in A\cap C$. Then $x\in A$ and $x\in C$.
Since $A\subseteq B$, we have $x\in B$. Thus $x\in B\cap C$, so
$A\cap C\subseteq B\cap C$.

Let $x\in C\cap A$. Then $x\in C$ and $x\in A$. Since $A\subseteq B$,
we have $x\in B$. Hence $x\in C\cap B$, and
$C\cap A\subseteq C\cap B$.
\end{proof}

\begin{proof}[Detailed Learning Proof]
\LRAProofBodyStart
Intersection requires both membership conditions at once. Starting from
$x\in A\cap C$ gives $x\in A$ and $x\in C$. The inclusion $A\subseteq B$
changes the first fact into $x\in B$, while the second fact remains unchanged.
Together these give $x\in B\cap C$. The right-side version keeps the common
set $C$ fixed and applies the same subset implication to the other coordinate.
\end{proof}
