\begin{theorem*}[Linear Order Induced by a Strict Linear Order]
If $<$ is a strict linear order on $S$ and
\[
x\leq y \;\Longleftrightarrow\; (x<y\lor x=y),
\]
then $\leq$ is a linear order on $S$.
\end{theorem*}
