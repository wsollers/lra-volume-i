\begin{toolkitbox}{Ordered Sets --- Quick Reference}

\hyperref[def:order-maximal-element]{Maximal element}   & No larger element in $S$                             \\

\hyperref[def:order-greatest-element]{Greatest element}    & Larger than all elements of $S$                     \\
\hyperref[def:order-preserving-map]{Order-preserving map}    & \hyperref[def:function]{Function} $f$ with $a \leq b \Rightarrow f(a) \leq' f(b)$                       \\
\hyperref[def:order-isomorphism]{Order isomorphism}   & \hyperref[def:bijective]{Bijective} \hyperref[def:function]{map} preserving and reflecting order \\
\hyperref[def:well-ordered-set]{Well-ordered set}

\hyperref[def:order-chain]{Chain} / \hyperref[def:order-antichain]{antichain}   & All \hyperref[def:comparable]{comparable} / none comparable                            \\
\hyperref[def:initial-segment]{Initial segment}     & Downward-closed \hyperref[def:subset]{subset}                                      \\
\bottomrule

\begin{toolkitbox}{Lattices --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[def:lattice]{Lattice}               & Poset with $\sup\{a,b\}$ and $\inf\{a,b\}$ for all $a,b$   \\
\hyperref[def:lattice]{Join / Meet}           & $a \vee b = \sup\{a,b\}$; $a \wedge b = \inf\{a,b\}$        \\
\hyperref[def:complete-lattice]{Complete lattice}      & Every subset has sup and inf                                \\
\hyperref[def:distributive-lattice]{Distributive lattice}  & $a \wedge (b \vee c) = (a \wedge b) \vee (a \wedge c)$      \\
\hyperref[def:boolean-lattice]{Boolean lattice}       & Complemented distributive lattice                           \\
\hyperref[def:lattice]{$\mathcal{P}(X)$}      & Complete Boolean lattice under $\subseteq$                  \\
\hyperref[def:boolean-lattice]{$\sigma$-algebras}     & Sub-lattices of $\mathcal{P}(X)$ closed under countable ops  \\
\hyperref[thm:knaster-tarski]{Knaster--Tarski}       & Every monotone map on a complete lattice has a fixed point   \\
\bottomrule
\end{tabular}
\end{toolkitbox}

\begin{toolkitbox}{Induced Orders and Order Embeddings --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Name}  & \textbf{Meaning} \\
\midrule
\hyperref[def:induced-order]{Induced preorder}     & $x \leq_f y \;\Leftrightarrow\; f(x) \leq' f(y)$; always a preorder  \\
\hyperref[prop:induced-poset]{Induced partial order}  & Induced order is a partial order iff $f$ is injective              \\
\hyperref[def:f-indist]{$f$-indistinguishable}  & $x \sim_f y$: both $x \leq_f y$ and $y \leq_f x$                   \\
\hyperref[def:order-embedding]{Order embedding}      & Injective, order-preserving, and order-reflecting                     \\
\hyperref[prop:embedding-iso]{Embedding vs.\ isomorphism}  & Embedding is iso onto image; isomorphism is surjective too \\
\hyperref[def:suborder]{Suborder}             & Restriction of $\leq'$ to a subset $S \subseteq B$                   \\
\midrule
\multicolumn{2}{l}{\textit{Key results:}} \\
\midrule
\multicolumn{2}{l}{\hyperref[prop:induced-poset]{$\leq_f$ preorder: always. \quad Antisymmetry: iff $f$ injective.}}       \\
\multicolumn{2}{l}{\hyperref[prop:embedding-iso]{$f$ order embedding $\Rightarrow$ $(A,\leq_f) \cong (f(A), \leq')$.}} \\
\bottomrule
\end{tabular}
