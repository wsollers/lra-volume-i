\begin{theorem*}[Distributive Laws for Propositional Logic]
For all \(\varphi,\psi,\chi\in\WFF\),
\[
\varphi\land(\psi\lor\chi)
\equiv
(\varphi\land\psi)\lor(\varphi\land\chi),
\]
and
\[
\varphi\lor(\psi\land\chi)
\equiv
(\varphi\lor\psi)\land(\varphi\lor\chi).
\]
\end{theorem*}
