\begin{theorem*}[De Morgan Laws for Propositional Logic]
For all \(\varphi,\psi\in\WFF\),
\[
\neg(\varphi\land\psi)
\equiv
\neg\varphi\lor\neg\psi,
\]
and
\[
\neg(\varphi\lor\psi)
\equiv
\neg\varphi\land\neg\psi.
\]
\end{theorem*}
