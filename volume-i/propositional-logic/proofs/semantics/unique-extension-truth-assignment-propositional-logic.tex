\label{prf:unique-extension-truth-assignment-propositional-logic}

\noindent\hyperref[thm:unique-extension-truth-assignment-propositional-logic]{$\leftarrow$ Back to Theorem (Unique Extension of a Truth Assignment) in Notes}

\begin{theorem*}[Unique Extension of a Truth Assignment]
Every truth assignment $v:\Prop\to\mathbb{B}$ extends uniquely to a function
\[
\widehat v:\WFF\to\mathbb{B}
\]
satisfying the following clauses:
\begin{enumerate}
  \item $\widehat v(P)=v(P)$ for every $P\in\Prop$.
  \item $\widehat v(\neg\varphi)=\True$ if and only if
        $\widehat v(\varphi)=\False$.
  \item $\widehat v((\varphi\land\psi))=\True$ if and only if
        $\widehat v(\varphi)=\True$ and $\widehat v(\psi)=\True$.
  \item $\widehat v((\varphi\lor\psi))=\True$ if and only if
        $\widehat v(\varphi)=\True$ or $\widehat v(\psi)=\True$.
  \item $\widehat v((\varphi\to\psi))=\True$ if and only if
        $\widehat v(\varphi)=\False$ or $\widehat v(\psi)=\True$.
  \item $\widehat v((\varphi\leftrightarrow\psi))=\True$ if and only if
        $\widehat v(\varphi)=\widehat v(\psi)$.
\end{enumerate}
\end{theorem*}
