\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Subformula)}},
  fonttitle=\small\bfseries]
\label{def:subformula}
The set $\mathrm{Sub}(\varphi)$ of \emph{subformulas} of a wff $\varphi$ is
defined recursively:
\begin{enumerate}
  \item If $\varphi$ is atomic, then
        $\mathrm{Sub}(\varphi)=\{\varphi\}$.
  \item If $\varphi=\neg\psi$, then
        $\mathrm{Sub}(\varphi)=\{\varphi\}\cup\mathrm{Sub}(\psi)$.
  \item If $\varphi=(\psi\circ\chi)$, then
        $\mathrm{Sub}(\varphi)
        =\{\varphi\}\cup\mathrm{Sub}(\psi)\cup\mathrm{Sub}(\chi)$.
\end{enumerate}
A \emph{proper subformula} is a subformula other than $\varphi$ itself.
\end{tcolorbox}

\begin{remark*}[Interpretation]
Subformulas are the formulas that occur at nodes of the parse tree. They are the
syntactic parts visible inside a formula.
\end{remark*}
