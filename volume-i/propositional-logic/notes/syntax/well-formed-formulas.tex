\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Well-Formed Formula)}},
  fonttitle=\small\bfseries]
\label{def:well-formed-formula}
The set $\mathsf{WFF}$ of \emph{well-formed formulas} is defined recursively:
\begin{enumerate}
  \item Every propositional variable is a wff.
  \item If $\varphi$ is a wff, then $\neg\varphi$ is a wff.
  \item If $\varphi,\psi$ are wffs and
        $\circ\in\{\land,\lor,\to,\leftrightarrow\}$, then
        $(\varphi\circ\psi)$ is a wff.
  \item Nothing else is a wff.
\end{enumerate}
\end{tcolorbox}

\begin{remark*}[Interpretation]
The first three clauses generate formulas. The closure clause says that a string
is a formula only if it is obtained by finitely many applications of those
clauses.
\end{remark*}
