\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Unique Readability for Propositional Logic)}},
  fonttitle=\small\bfseries]
\label{thm:unique-readability-propositional-logic}
Every well-formed formula has exactly one of the following forms:
\begin{enumerate}
  \item an atomic formula $P\in\Prop$;
  \item $\neg\psi$ for a unique wff $\psi$;
  \item $(\psi\circ\chi)$ for a unique binary connective $\circ$ and unique
        wffs $\psi,\chi$.
\end{enumerate}

\noindent\hyperref[prf:unique-readability-propositional-logic]{Proof location.}
\end{tcolorbox}

\begin{remark*}[Interpretation]
Unique readability guarantees that recursive definitions and structural
induction are well-defined. A proof is supplied later in the chapter proof
section.
\end{remark*}
