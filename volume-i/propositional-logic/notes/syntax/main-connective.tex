\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Main Connective)}},
  fonttitle=\small\bfseries]
\label{def:main-connective}
The \emph{main connective} of a molecular formula is the connective introduced
at the final formation step:
\begin{itemize}
  \item for $\neg\varphi$, the main connective is $\neg$;
  \item for $(\varphi\circ\psi)$, the main connective is $\circ$.
\end{itemize}
Atomic formulas have no main connective.
\end{tcolorbox}

\begin{remark*}[Interpretation]
The main connective identifies the outermost syntactic operation. It is the
first connective used when analyzing a formula from the outside inward.
\end{remark*}
