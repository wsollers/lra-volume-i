\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Formula Depth)}},
  fonttitle=\small\bfseries]
\label{def:formula-depth-propositional-logic}
The \emph{depth} of a wff $\varphi$ is defined recursively:
\begin{enumerate}
  \item If $\varphi$ is atomic, then $\mathrm{depth}(\varphi)=0$.
  \item If $\varphi=\neg\psi$, then
        $\mathrm{depth}(\varphi)=\mathrm{depth}(\psi)+1$.
  \item If $\varphi=(\psi\circ\chi)$, then
        \[
        \mathrm{depth}(\varphi)
        =\max\{\mathrm{depth}(\psi),\mathrm{depth}(\chi)\}+1.
        \]
\end{enumerate}
\end{tcolorbox}

\begin{remark*}[Interpretation]
Formula depth measures structural complexity. It is the height of the formation
tree, with atomic formulas at depth zero.
\end{remark*}
