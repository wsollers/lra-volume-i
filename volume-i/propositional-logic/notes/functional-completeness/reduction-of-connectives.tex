\begin{definition}[Connective Reduction]
\label{def:connective-reduction-propositional-logic}
A \emph{connective reduction} replaces a formula using one connective basis by
a logically equivalent formula using a smaller or different connective basis.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{Reduction}_{B\to C}(\varphi,\psi)
\Longleftrightarrow
\varphi\equiv\psi
\quad\text{and}\quad
\psi\in\WFF_C.
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\(\operatorname{Reduction}_{B\to C}(\varphi,\psi)\) means that \(\psi\) is a
translation of \(\varphi\) into the connective basis \(C\) without changing truth
behavior.
\end{remark*}

\begin{remark*}[Interpretation]
Reduction shows that some primitive connectives are optional. If a connective is
definable from the remaining basis, then formulas using that connective can be
translated away.
\end{remark*}

\begin{remark*}[Exposition]
The standard reductions are:
\[
\varphi\to\psi\equiv\neg\varphi\lor\psi,
\]
\[
\varphi\leftrightarrow\psi
\equiv
(\varphi\to\psi)\land(\psi\to\varphi),
\]
\[
\varphi\lor\psi\equiv\neg(\neg\varphi\land\neg\psi),
\qquad
\varphi\land\psi\equiv\neg(\neg\varphi\lor\neg\psi).
\]
Each replacement preserves logical equivalence and may be performed inside a
larger formula context.
\end{remark*}

\begin{remark*}[Exposition]
Reducing a connective basis can make formulas longer. Functional completeness is
therefore an expressive result, not an efficiency result. A small basis may be
powerful enough to express every Boolean function while still being awkward for
ordinary human-readable formulas.
\end{remark*}
