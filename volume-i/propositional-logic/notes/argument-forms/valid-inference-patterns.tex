\subsubsection*{Core Valid Inference Patterns}

\begin{remark*}[Working vocabulary]
This topic proves the semantic validity of the standard propositional inference
patterns used throughout later proof work.
\end{remark*}

  title={\small\textbf{Theorem (Modus Ponens is Valid)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Modus Ponens is Valid]
\label{thm:modus-ponens-validity-propositional-logic}
The argument form
\[
\varphi,\qquad \varphi\to\psi \quad\therefore\quad \psi
\]
is valid.

\noindent\hyperref[prf:modus-ponens-validity-propositional-logic]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
(\forall\varphi,\psi\in\WFF)\;\operatorname{ValidArgumentForm}(\{\varphi,\varphi\to\psi\},\psi).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
Whenever \(\varphi\) and \(\varphi\to\psi\) are both true, \(\psi\) must be true.
\end{remark*}

\begin{remark*}[Interpretation]
Modus ponens is the basic rule for using a conditional whose antecedent has
already been established.
\end{remark*}

\begin{theorem}[Modus Tollens is Valid]
\label{thm:modus-tollens-validity-propositional-logic}
The argument form
\[
\varphi\to\psi,\qquad \neg\psi \quad\therefore\quad \neg\varphi
\]
is valid.

\noindent\hyperref[prf:modus-tollens-validity-propositional-logic]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
(\forall\varphi,\psi\in\WFF)\;\operatorname{ValidArgumentForm}(\{\varphi\to\psi,\neg\psi\},\neg\varphi).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
If \(\varphi\) would imply \(\psi\), and \(\psi\) is false, then \(\varphi\) must be false.
\end{remark*}

\begin{remark*}[Interpretation]
Modus tollens is the semantic form of denying an antecedent by denying the
conditional's consequent.
\end{remark*}

\begin{theorem}[Disjunctive Syllogism is Valid]
\label{thm:disjunctive-syllogism-validity-propositional-logic}
The argument form
\[
\varphi\lor\psi,\qquad \neg\varphi \quad\therefore\quad \psi
\]
is valid.

\noindent\hyperref[prf:disjunctive-syllogism-validity-propositional-logic]{Go to proof.}
\end{theorem}

\[
(\forall\varphi,\psi\in\WFF)\;\operatorname{ValidArgumentForm}(\{\varphi\lor\psi,\neg\varphi\},\psi).

\begin{remark*}[Predicate reading]
If at least one of \(\varphi\) or \(\psi\) is true and \(\varphi\) is false, then \(\psi\) must be true.
\end{remark*}

\begin{remark*}[Interpretation]
Disjunctive syllogism eliminates one disjunct after its negation is known.
\end{remark*}

  fonttitle=\small\bfseries]
\begin{theorem}[Constructive Dilemma is Valid]
\label{thm:constructive-dilemma-validity-propositional-logic}
The argument form
\[
\varphi\to\chi,\qquad \psi\to\theta,\qquad \varphi\lor\psi
\quad\therefore\quad
\chi\lor\theta
\]
is valid.

\noindent\hyperref[prf:constructive-dilemma-validity-propositional-logic]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
(\forall\varphi,\psi,\chi,\theta\in\WFF)
\operatorname{ValidArgumentForm}(\{\varphi\to\chi,\psi\to\theta,\varphi\lor\psi\},\chi\lor\theta).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
If one of two antecedents holds, and each antecedent has its own consequence,
then at least one corresponding consequence holds.
\end{remark*}

\begin{remark*}[Interpretation]
Constructive dilemma is a proof-by-cases pattern with conditionals attached to
both cases.
\end{remark*}
