\begin{theorem}[Semantic Classification by Normal Forms]
\label{thm:normal-form-semantic-classification-propositional-logic}
Normal forms give finite certificates for semantic classification. For any
formula \(\varphi\in\WFF\):
\begin{enumerate}
  \item \(\varphi\) is
  \hyperref[def:satisfiable-formula-propositional-logic]{satisfiable} if and
  only if some
  \hyperref[def:disjunctive-normal-form-propositional-logic]{DNF} formula
  equivalent to \(\varphi\) has a satisfiable conjunction term.
  \item \(\varphi\) is a
  \hyperref[def:contradiction-propositional-logic]{contradiction} if and only
  if some DNF formula equivalent to \(\varphi\) has no satisfiable conjunction
  term.
  \item \(\varphi\) is a
  \hyperref[def:tautology-propositional-logic]{tautology} if and only if some
  \hyperref[def:conjunctive-normal-form-propositional-logic]{CNF} formula
  equivalent to \(\varphi\) has no falsifiable clause.
  \item \(\varphi\) is a
  \hyperref[def:contingency-propositional-logic]{contingency} if and only if it
  is satisfiable and not a tautology.
\end{enumerate}
\noindent\hyperref[prf:normal-form-semantic-classification-propositional-logic]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
\varphi\text{ is satisfiable}
&\Longleftrightarrow
\text{some equivalent DNF term is satisfiable},\\
\varphi\text{ is a contradiction}
&\Longleftrightarrow
\text{no equivalent DNF term is satisfiable},\\
\varphi\text{ is a tautology}
&\Longleftrightarrow
\text{no equivalent CNF clause is falsifiable}.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The truth-table method classifies \(\varphi\) by evaluating every row directly.
The normal-form method first rewrites \(\varphi\) into an equivalent DNF or CNF
formula and then checks the finite terms or clauses. Canonical normal forms
connect the two approaches: canonical DNF is built from true rows, and
canonical CNF is built from false rows.
\end{remark*}

\begin{remark*}[Exposition]
DNF is naturally witness-producing. A satisfiable term gives an explicit way
to make the whole disjunction true. CNF is naturally suited for clause
checking: each clause is a local obstruction to falsifying the whole
conjunction. This clause-centered view is also the starting point for later
resolution methods.
\end{remark*}
