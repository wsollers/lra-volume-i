\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Satisfaction)}},
  fonttitle=\small\bfseries]
\label{def:satisfaction-propositional-logic}
Let $v$ be a truth assignment and let $\varphi\in\WFF$. The assignment
$v$ \emph{satisfies} $\varphi$, written
\[
v\models\varphi,
\]
if $\widehat v(\varphi)=\True$.

If $\Gamma\subseteq\WFF$, then $v$ \emph{satisfies} $\Gamma$, written
$v\models\Gamma$, if $v\models\gamma$ for every $\gamma\in\Gamma$.
\end{tcolorbox}

\begin{remark*}[Interpretation]
The notation $v\models\varphi$ says that $\varphi$ is true under the semantic
interpretation determined by $v$. Satisfaction for a set of formulas means
simultaneous satisfaction of every formula in the set.
\end{remark*}
