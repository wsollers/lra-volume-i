\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Truth Assignment)}},
  fonttitle=\small\bfseries]
\label{def:truth-assignment-propositional-logic}
A \emph{truth assignment} for propositional logic is a function
\[
v:\Prop\to\mathbb{B}.
\]
Thus a truth assignment assigns either $\False$ or $\True$ to each
propositional variable.
\end{tcolorbox}

\begin{remark*}[Interpretation]
A truth assignment fixes the truth values of atomic formulas. The recursive
semantics of the connectives then determines the truth value of each molecular
formula.
\end{remark*}
