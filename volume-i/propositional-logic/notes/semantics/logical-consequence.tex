\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Logical Consequence)}},
  fonttitle=\small\bfseries]
\label{def:logical-consequence-propositional-logic}
Let $\Gamma\subseteq\WFF$ and let $\varphi\in\WFF$. The formula $\varphi$ is a
\emph{logical consequence} of $\Gamma$, written
\[
\Gamma\models\varphi,
\]
if every truth assignment that satisfies $\Gamma$ also satisfies $\varphi$.
\end{tcolorbox}

\begin{remark*}[Interpretation]
The assertion $\Gamma\models\varphi$ says that there is no truth assignment
under which every member of $\Gamma$ is true while $\varphi$ is false. When
$\Gamma=\emptyset$, this is exactly the assertion that $\varphi$ is a
tautology.
\end{remark*}
