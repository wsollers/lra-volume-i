\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Tautology)}},
  fonttitle=\small\bfseries]
\label{def:tautology-propositional-logic}
A formula $\varphi$ is a \emph{tautology}, written
\[
\models\varphi,
\]
if $v\models\varphi$ for every truth assignment $v$.
\end{tcolorbox}

\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Contradiction)}},
  fonttitle=\small\bfseries]
\label{def:contradiction-propositional-logic}
A formula $\varphi$ is a \emph{contradiction} if $v\not\models\varphi$ for
every truth assignment $v$.
\end{tcolorbox}

\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Satisfiable Formula)}},
  fonttitle=\small\bfseries]
\label{def:satisfiable-formula-propositional-logic}
A formula $\varphi$ is \emph{satisfiable} if $v\models\varphi$ for at least one
truth assignment $v$.
\end{tcolorbox}

\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Contingency)}},
  fonttitle=\small\bfseries]
\label{def:contingency-propositional-logic}
A formula $\varphi$ is a \emph{contingency} if it is satisfiable but not a
tautology.
\end{tcolorbox}

\begin{remark*}[Interpretation]
Tautologies are true in all rows of the relevant truth table, contradictions
are true in no rows, and contingencies are true in some rows but not all rows.
\end{remark*}
