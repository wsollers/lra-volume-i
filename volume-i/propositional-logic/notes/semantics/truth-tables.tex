\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Truth Table)}},
  fonttitle=\small\bfseries]
\label{def:truth-table-propositional-logic}
Let $\varphi$ be a formula whose propositional variables are among
$P_1,\ldots,P_n$. A \emph{truth table} for $\varphi$ is a table with one row
for each function $a:\{P_1,\ldots,P_n\}\to\mathbb{B}$ and with a final column
recording the value of $\widehat v(\varphi)$ for any truth assignment $v$
extending $a$.
\end{tcolorbox}

\begin{remark*}[Interpretation]
A truth table is finite because a formula contains only finitely many
propositional variables. If $\varphi$ uses $n$ distinct variables, then there
are $2^n$ rows.
\end{remark*}

\begin{center}
\small
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{cc|ccccc}
\toprule
$P$ & $Q$ & $\neg P$ & $P\land Q$ & $P\lor Q$ & $P\to Q$ & $P\leftrightarrow Q$ \\
\midrule
$\True$ & $\True$ & $\False$ & $\True$ & $\True$ & $\True$ & $\True$ \\
$\True$ & $\False$ & $\False$ & $\False$ & $\True$ & $\False$ & $\False$ \\
$\False$ & $\True$ & $\True$ & $\False$ & $\True$ & $\True$ & $\False$ \\
$\False$ & $\False$ & $\True$ & $\False$ & $\False$ & $\True$ & $\True$ \\
\bottomrule
\end{tabular}
\end{center}
