\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Logical Equivalence)}},
  fonttitle=\small\bfseries]
\label{def:logical-equivalence-propositional-logic}
Formulas $\varphi,\psi\in\WFF$ are \emph{logically equivalent}, written
\[
\varphi\equiv\psi,
\]
if $\widehat v(\varphi)=\widehat v(\psi)$ for every truth assignment $v$.
\end{tcolorbox}

\begin{remark*}[Interpretation]
Logically equivalent formulas have the same truth table column under every
assignment of their variables. Equivalently, $\{\varphi\}\models\psi$ and
$\{\psi\}\models\varphi$.
\end{remark*}
