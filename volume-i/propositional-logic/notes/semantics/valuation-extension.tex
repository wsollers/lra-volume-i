\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Unique Extension of a Truth Assignment)}},
  fonttitle=\small\bfseries]
\label{thm:unique-extension-truth-assignment-propositional-logic}
Every truth assignment $v:\Prop\to\mathbb{B}$ extends uniquely to a function
\[
\widehat v:\WFF\to\mathbb{B}
\]
satisfying the following clauses:
\begin{enumerate}
  \item $\widehat v(P)=v(P)$ for every $P\in\Prop$.
  \item $\widehat v(\neg\varphi)=\True$ if and only if
        $\widehat v(\varphi)=\False$.
  \item $\widehat v((\varphi\land\psi))=\True$ if and only if
        $\widehat v(\varphi)=\True$ and $\widehat v(\psi)=\True$.
  \item $\widehat v((\varphi\lor\psi))=\True$ if and only if
        $\widehat v(\varphi)=\True$ or $\widehat v(\psi)=\True$.
  \item $\widehat v((\varphi\to\psi))=\True$ if and only if
        $\widehat v(\varphi)=\False$ or $\widehat v(\psi)=\True$.
  \item $\widehat v((\varphi\leftrightarrow\psi))=\True$ if and only if
        $\widehat v(\varphi)=\widehat v(\psi)$.
\end{enumerate}

\noindent\hyperref[prf:unique-extension-truth-assignment-propositional-logic]{Proof location.}
\end{tcolorbox}

\begin{remark*}[Interpretation]
The theorem justifies writing $v(\varphi)$ for the value of a formula
$\varphi$, even though $v$ is initially specified only on propositional
variables. Unique readability guarantees that the recursive clauses apply in
exactly one way.
\end{remark*}
