\begin{tcolorbox}[colback=structurebox, colframe=structureborder, arc=2pt,
  left=8pt, right=8pt, top=6pt, bottom=6pt,
  title={\small\textbf{Architecture Presentation: The Four Layers of Logic}},
  fonttitle=\small\bfseries]
The basic architecture of logic separates four kinds of questions:
\[
\begin{array}{c}
\text{Syntax}\\
\Downarrow\\
\text{Semantics}\\
\Downarrow\\
\text{Proof Theory}\\
\Downarrow\\
\text{Metatheory}
\end{array}
\]

\textbf{Syntax} fixes the formal symbols and formation rules of the language.
It determines which expressions count as formulas.

\textbf{Semantics} assigns meaning to the formulas supplied by syntax. It
explains what it is for a formula to hold in a semantic setting.

\textbf{Proof theory} studies formal derivations. It explains how formulas
are obtained from assumptions by rules of inference.

\textbf{Metatheory} studies the logical system itself. It compares the
semantic relation of truth-preservation with the proof-theoretic relation of
derivability.
\end{tcolorbox}

\begin{remark*}[Why logic has layers]
Different questions belong to different layers. Asking whether an expression
is legal is a syntactic question. Asking whether a legal expression is true is
a semantic question. Asking whether a conclusion can be derived is a
proof-theoretic question. Asking why proof and truth agree is a metatheoretic
question.
\end{remark*}

\begin{remark*}[Reuse throughout mathematics]
The same architecture reappears throughout the logic sequence. Predicate
logic, axiom systems, model theory, and type theory all distinguish formal
expressions, semantic interpretation, proof systems, and metatheoretic
properties of those systems.
\end{remark*}

\begin{remark*}[Roadmap through Chapter 1]
The Propositional Logic chapter follows this order. The syntax sections build
the formal language. The semantics sections explain meaning and consequence.
The proof systems section introduces derivations. The metatheory section then
studies the relationship between semantic consequence and formal proof.
\end{remark*}

\begin{definition}[Syntax]
\label{def:propositional-logic-syntax}
\emph{Syntax} is the collection of formal symbols and formation rules that
determine which expressions belong to a language.
\end{definition}

\begin{remark*}[Interpretation]
Syntax is the grammar layer of logic. It decides whether an expression is
well-formed before any question of meaning, truth, or proof is considered.
\end{remark*}

\begin{definition}[Semantics]
\label{def:propositional-logic-semantics}
\emph{Semantics} assigns meaning to syntactically valid expressions.
\end{definition}

\begin{remark*}[Interpretation]
Semantics begins only after syntax has supplied well-formed expressions. It
explains what those expressions mean and how their truth is evaluated within
an appropriate semantic setting.
\end{remark*}

\begin{definition}[Proof Theory]
\label{def:propositional-logic-proof-theory}
\emph{Proof theory} studies formal derivations and rules of inference.
\end{definition}

\begin{remark*}[Interpretation]
Proof theory treats reasoning as a formal activity. It asks which conclusions
can be derived from which assumptions using specified rules.
\end{remark*}

\begin{definition}[Metatheory]
\label{def:propositional-logic-metatheory}
\emph{Metatheory} studies properties of a logical system itself.
\end{definition}

\begin{remark*}[Interpretation]
Metatheory looks at the logical system from outside the system. In this
chapter, its central role is to compare the semantic notion of consequence
with the proof-theoretic notion of derivation.
\end{remark*}
