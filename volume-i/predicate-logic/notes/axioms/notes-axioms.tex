\begin{topicbox}{Predicate-Logic Axiom Schemas}
\begin{exposition}
The predicate-logic development extends propositional logic with the following
quantifier axiom schemas. The side conditions prevent variable capture and
ensure that the displayed instances are well formed.
\end{exposition}
\end{topicbox}

\begin{axiombox}{Axiom Schema (Predicate Logic 1)}
\begin{axiom}[Universal Instantiation]
\label{ax:predicate-logic-universal-instantiation}
For every formula $\varphi(x)$ and every term $t$ that is free for $x$ in
$\varphi$,
\[
\forall x\,\varphi(x) \Longrightarrow \varphi(t).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Interpretation]
A universally quantified statement may be specialized to any admissible
instance.
\end{remark*}

\begin{axiombox}{Axiom Schema (Predicate Logic 2)}
\begin{axiom}[Quantifier Distribution]
\label{ax:predicate-logic-quantifier-distribution}
For formulas $\varphi$ and $\psi(x)$, if $x$ is not free in $\varphi$, then
\[
\forall x\,(\varphi \Longrightarrow \psi(x))
\Longrightarrow
\bigl(\varphi \Longrightarrow \forall x\,\psi(x)\bigr).
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Interpretation]
When a hypothesis does not depend on the quantified variable, a universal
conclusion may be drawn uniformly under that hypothesis.
\end{remark*}
