\begin{theorem*}[Valid Quantifier Distribution]
For all formulas \(\varphi,\psi\),
\[
\forall x\,(\varphi\land\psi)
\equiv
(\forall x\,\varphi)\land(\forall x\,\psi),
\]
\[
\exists x\,(\varphi\lor\psi)
\equiv
(\exists x\,\varphi)\lor(\exists x\,\psi).
\]
\end{theorem*}
