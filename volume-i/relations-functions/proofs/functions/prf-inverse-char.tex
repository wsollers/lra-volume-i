\begin{theorem*}[Characterization of Inverse Functions]
Let $f : A \to B$ be bijective. Then
\[
f^{-1} \circ f = \id_A
\quad\text{and}\quad
f \circ f^{-1} = \id_B.
\]
Conversely, a function admits an inverse iff it is bijective.
\end{theorem*}
