\begin{toolkitbox}{Relations --- Quick Reference}
\small
\begin{tabular}{l l l}
\toprule
\textbf{Concept} & \textbf{Meaning} & \textbf{Detail} \\
\midrule
Ordered pair     & $(a,b) := \{\{a\},\{a,b\}\}$; identity iff both coords equal  & \hyperref[def:ordered-pair]{$\downarrow$ Def} \\
Cartesian product & $A \times B$: all ordered pairs from $A$ and $B$             & \hyperref[def:cartesian-product-rel]{$\downarrow$ Def} \\

Range of $R$     & Outputs that occur in $R$                                     & \hyperref[def:relation-range]{$\downarrow$ Def} \\
Converse         & Reverse every ordered pair                                    & \hyperref[def:converse-relation]{$\downarrow$ Def} \\
Identity relation & Diagonal relation $\Delta_A$                                 & \hyperref[def:relation-identity]{$\downarrow$ Def} \\

\bottomrule
\end{tabular}
\end{toolkitbox}

\midrule

Universal relation  & $A\times B$                                                 & \hyperref[def:universal-relation]{$\downarrow$ Def} \\
Relation inclusion  & $R\subseteq S$ as set inclusion                             & \hyperref[def:relation-inclusion]{$\downarrow$ Def} \\
Relation equality   & Mutual relation inclusion                                   & \hyperref[def:relation-equality]{$\downarrow$ Def} \\
Union of relations  & Pair belongs to at least one relation                       & \hyperref[def:relation-union]{$\downarrow$ Def} \\

Difference          & Pair belongs to first relation but not second               & \hyperref[def:relation-difference]{$\downarrow$ Def} \\
Complement          & Pairs in $A\times B$ not in $R$                             & \hyperref[def:relation-complement]{$\downarrow$ Def} \\
Converse laws       & Converse reverses composition and preserves Boolean operations & \hyperref[thm:relation-converse-laws]{$\downarrow$ Thm} \\
Composition laws    & Associative; distributes over unions, not all Boolean operations & \hyperref[thm:relation-composition-boolean]{$\downarrow$ Thm} \\
One-to-many         & One input has two distinct outputs                          & \hyperref[def:one-to-many-relation]{$\downarrow$ Def} \\
Many-to-one         & Two distinct inputs share one output                        & \hyperref[def:many-to-one-relation]{$\downarrow$ Def} \\
Many-to-many        & One-to-many and many-to-one phenomena both occur            & \hyperref[def:many-to-many-relation]{$\downarrow$ Def} \\

\begin{toolkitbox}{Relational Properties --- Quick Reference}
\small
\begin{tabular}{l l l l}
\toprule
\textbf{Property} & \textbf{Formal definition} & \textbf{Canonical example} & \textbf{Detail} \\
\midrule
Reflexive      & $\forall a,\; (a,a)\in R$                           & $=$ on any set      & \hyperref[def:reflexive]{$\downarrow$ Def} \\
Irreflexive    & $\forall a,\; (a,a)\notin R$                        & $<$ on $\mathbb{N}$ & \hyperref[def:irreflexive]{$\downarrow$ Def} \\
Symmetric      & $(a,b)\in R \Rightarrow (b,a)\in R$                 & ``same age as''     & \hyperref[def:symmetric]{$\downarrow$ Def} \\
Antisymmetric  & $(a,b),(b,a)\in R \Rightarrow a=b$                  & $\leq$ on $\mathbb{N}$ & \hyperref[def:antisymmetric]{$\downarrow$ Def} \\
Asymmetric     & $(a,b)\in R \Rightarrow (b,a)\notin R$              & $<$ on $\mathbb{N}$ & \hyperref[def:asymmetric]{$\downarrow$ Def} \\
Transitive     & $(a,b),(b,c)\in R \Rightarrow (a,c)\in R$           & $\leq$ on $\mathbb{N}$ & \hyperref[def:transitive]{$\downarrow$ Def} \\
Total (Connex) & $(a,b)\in R \lor (b,a)\in R$                        & $\leq$ on $\mathbb{R}$ & \hyperref[def:total-rel]{$\downarrow$ Def} \\
\midrule
\multicolumn{4}{l}{\textit{Structural classes formed by combining properties:}} \\
\midrule
Equivalence    & Reflexive + Symmetric + Transitive                  & $=$ on any set      & \hyperref[def:equivalence-rel]{$\downarrow$ Def} \\
Preorder       & Reflexive + Transitive                              & $\leq$ on $\mathbb{N}$ & \hyperref[def:preorder]{$\downarrow$ Def} \\
Partial order  & Reflexive + Antisymmetric + Transitive              & $\subseteq$ on $\mathcal{P}(A)$ & \hyperref[def:partial-order]{$\downarrow$ Def} \\
Total order    & Partial order + Total                               & $\leq$ on $\mathbb{R}$ & \hyperref[def:total-order]{$\downarrow$ Def} \\
\bottomrule
\end{tabular}
\end{toolkitbox}
